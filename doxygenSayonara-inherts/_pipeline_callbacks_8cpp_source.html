53 void PipelineCallbacks::pad_added_handler(GstElement* src, GstPad* new_src_pad, gpointer data) {
54  Q_UNUSED(src)
55 
56  GstElement* element;
57  GstPad* sink_pad;
58  GstPadLinkReturn pad_link_return;
59 
60  element = static_cast<GstElement*>(data);
61  if(!element){
62  return;
63  }
64 
65  sink_pad = gst_element_get_static_pad(element, "sink");
66  if(!sink_pad){
67  return;
68  }
69 
70  if(gst_pad_is_linked(sink_pad)) {
71  return;
72  }
73 
74  pad_link_return = gst_pad_link(new_src_pad, sink_pad);
75 
76  if(pad_link_return != GST_PAD_LINK_OK) {
77  sp_log(Log::Error) << "Dynamic pad linking: Cannot link pads";
78 
79  switch(pad_link_return){
80  case GST_PAD_LINK_WRONG_HIERARCHY:
81  sp_log(Log::Error) << "Cause: Wrong hierarchy";
82  break;
83  case GST_PAD_LINK_WAS_LINKED:
84  sp_log(Log::Error) << "Cause: Pad was already linked";
85  break;
86  case GST_PAD_LINK_WRONG_DIRECTION:
87  sp_log(Log::Error) << "Cause: Pads have wrong direction";
88  break;
89  case GST_PAD_LINK_NOFORMAT:
90  sp_log(Log::Error) << "Cause: Pads have incompatible format";
91  break;
92  case GST_PAD_LINK_NOSCHED:
93  sp_log(Log::Error) << "Cause: Pads cannot cooperate scheduling";
94  break;
95  case GST_PAD_LINK_REFUSED:
96  default:
97  sp_log(Log::Error) << "Cause: Refused because of different reason";
98  break;
99  }
100  }
101 }
