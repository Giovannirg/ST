218 bool EngineHandler::configure_connections(Engine* old_engine, Engine* new_engine) {
219  if(!old_engine && !new_engine) return false;
220  if(old_engine == new_engine) return false;
221 
222  if(old_engine) {
223  disconnect(old_engine, &Engine::sig_track_ready, this, &EngineHandler::sl_track_ready_changed);
224  disconnect(old_engine, &Engine::sig_md_changed, this, &EngineHandler::sl_md_changed);
225  disconnect(old_engine, &Engine::sig_pos_changed_ms, this, &EngineHandler::sl_pos_changed_ms);
226  disconnect(old_engine, &Engine::sig_pos_changed_s, this, &EngineHandler::sl_pos_changed_s);
227  disconnect(old_engine, &Engine::sig_dur_changed, this, &EngineHandler::sl_dur_changed);
228  disconnect(old_engine, &Engine::sig_br_changed, this, &Engine::sig_br_changed);
229  disconnect(old_engine, &Engine::sig_track_finished, this, &EngineHandler::sl_track_finished);
230  disconnect(old_engine, &Engine::sig_buffering, this, &EngineHandler::sl_buffer_state_changed);
231  disconnect(old_engine, &Engine::sig_cover_changed, this, &EngineHandler::sig_cover_changed);
232  }
233 
234  if(new_engine) {
235  connect(new_engine, &Engine::sig_track_ready, this, &EngineHandler::sl_track_ready_changed);
236  connect(new_engine, &Engine::sig_md_changed, this, &EngineHandler::sl_md_changed);
237  connect(new_engine, &Engine::sig_pos_changed_ms, this, &EngineHandler::sl_pos_changed_ms);
238  connect(new_engine, &Engine::sig_pos_changed_s, this, &EngineHandler::sl_pos_changed_s);
239  connect(new_engine, &Engine::sig_dur_changed, this, &EngineHandler::sl_dur_changed);
240  connect(new_engine, &Engine::sig_br_changed, this, &Engine::sig_br_changed);
241  connect(new_engine, &Engine::sig_track_finished, this, &EngineHandler::sl_track_finished);
242  connect(new_engine, &Engine::sig_buffering, this, &EngineHandler::sl_buffer_state_changed);
243  connect(new_engine, &Engine::sig_cover_changed, this, &EngineHandler::sig_cover_changed);
244  }
245 
246  return true;
247 }

249 Engine* EngineHandler::get_engine(EngineName name)
250 {
251  for(Engine* e : _engines){
252  if(e && e->get_name() == name){
253  return e;
254  }
255  }
256 
257  if(name == EngineName::PlaybackEngine){
258  PlaybackEngine* pb_engine = new PlaybackEngine();
259  if(pb_engine->init()){
260  _engines << static_cast<Engine*>(pb_engine);
261  connect(pb_engine, &PlaybackEngine::sig_data, this, &EngineHandler::new_data);
262  return pb_engine;
263  }
264  }
265 
266  else if(name == EngineName::ConvertEngine){
267  ConvertEngine* cvt_engine = new ConvertEngine();
268  if(cvt_engine->init()){
269  _engines << static_cast<Engine*>(cvt_engine);
270  return cvt_engine;
271 
272  }
273  }
274 
275  return nullptr;
276 }
