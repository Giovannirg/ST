258 bool SoundcloudJsonParser::parse_playlist(ArtistList& artists, Album& album, MetaDataList& v_md, QJsonObject object)
259 {
260  Artist pl_artist;
261 
262  get_int("id", object, album.id);
263  get_string("title", object, album.name);
264  get_string("artwork_url", object, album.cover_download_url);
265 
266  int num_songs;
267  if(get_int("track_count", object, num_songs)){
268  album.num_songs = num_songs;
269  }
270 
271  int length;
272  if(get_int("duration", object, length)){
273  album.length_sec = length / 1000;
274  }
275 
276  QJsonObject artist_object;
277  if(get_object("user", object, artist_object)){
278  parse_artist(pl_artist, artist_object);
279  if(!artists.contains(pl_artist.id) && pl_artist.id > 0){
280  artists << pl_artist;
281  }
282  }
283 
284  QJsonArray track_array;
285  if(get_array("tracks", object, track_array)){
286  ArtistList tmp_artists;
287  MetaDataList v_md_tmp;
288  parse_track_list(tmp_artists, v_md_tmp, track_array);
289  for(const Artist& tmp_artist : tmp_artists){
290  if(!artists.contains(tmp_artist.id)){
291  artists << tmp_artist;
292  }
293  }
294 
295  for(const MetaData& md : v_md_tmp){
296  if(!v_md.contains(md.id)){
297  v_md << md;
298  }
299  }
300  }
301 
302  QString permalink, purchase_url;
303  if(get_string("permalink", object, permalink)){
304  album.add_custom_field(permalink, tr("Permalink Url"), create_link("Soundcloud", permalink));
305  }
306 
307  if(get_string("purchase_url", object, purchase_url)){
308  album.add_custom_field(purchase_url, tr("Purchase Url"), create_link(purchase_url, purchase_url));
309  }
310 
311  QString album_name = album.name;
312 
313  for(int i=0; i<v_md.size(); i++){
314  v_md[i].track_num = i+1;
315  v_md[i].album = album.name;
316  v_md[i].album_id = album.id;
317 
318  if(v_md[i].artist_id != pl_artist.id && pl_artist.id > 0 && v_md[i].artist_id > 0){
319  v_md[i].album += " (by " + pl_artist.name + ")";
320  album_name = album.name + " (by " + pl_artist.name + ")";
321  }
322 
323  if(!album.cover_download_url.isEmpty()){
324  v_md[i].cover_download_url = album.cover_download_url;
325  }
326  }
327 
328  album.name = album_name;
329 
330  for(const Artist& artist : artists){
331  album.artists << artist.name;
332  }
333 
334  return (album.id > 0);
335 }
