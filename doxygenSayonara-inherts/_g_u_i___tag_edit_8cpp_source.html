628 void GUI_TagEdit::apply_tag(int idx)
629 {
630  if(!check_idx(idx)) {
631  return;
632  }
633 
634  QMap<Tag, ReplacedString> tag_cap_map = _m->tag_expression.get_tag_val_map();
635  MetaData md =_m->tag_edit->get_metadata(idx);
636 
637  for(const QString& tag : tag_cap_map.keys()){
638  ReplacedString cap = tag_cap_map[tag];
639 
640  if(tag.compare(TAG_TITLE) == 0){
641  md.title = cap;
642  }
643 
644  else if(tag.compare(TAG_ALBUM) == 0){
645  md.album = cap;
646  }
647 
648  else if(tag.compare( TAG_ARTIST) == 0){
649  md.artist = cap;
650  }
