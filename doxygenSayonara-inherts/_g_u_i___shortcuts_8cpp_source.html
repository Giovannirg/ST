120 void GUI_Shortcuts::sequence_entered()
121 {
122  GUI_ShortcutEntry* entry = static_cast<GUI_ShortcutEntry*>(sender());
123  QList<QKeySequence> sequences = entry->get_sequences();
124 
125  for(GUI_ShortcutEntry* lst_entry : _entries)
126  {
127  if(lst_entry == entry){
128  continue;
129  }
130 
131  QList<QKeySequence> saved_sequences = lst_entry->get_sequences();
132  for(const QKeySequence& seq1 : sequences)
133  {
134  QString seq1_str = seq1.toString(QKeySequence::NativeText);
135 
136  for(const QKeySequence& seq2 : saved_sequences)
137  {
138  QString seq2_str = seq2.toString(QKeySequence::NativeText);
139  if(seq1_str == seq2_str && !seq1_str.isEmpty()){
140  entry->show_sequence_error();
141  break;
142  }
143  }
144  }
145  }
