40 int parse_length_s(const QString& str)
41 {
42  QStringList lst = str.split(":");
43  int h=0;
44  int m=0;
45  int s=0;
46 
47  if(lst.size() == 3)
48  {
49  h = lst[0].toInt();
50  m = lst[1].toInt();
51  s = lst[2].split(".").first().toInt();
52  }
53 
54  if(lst.size() == 2)
55  {
56  m = lst[0].toInt();
57  s = lst[1].split(".").first().toInt();
58  }
59 
60  if(lst.size() == 1)
61  {
62  s = lst[0].split(".").first().toInt();
63  }
64 
65  return (h * 3600 + m * 60 + s);
66 }
