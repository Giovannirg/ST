41 bool colFromString(QString str, QColor& c) {
42  QStringList colors = str.split(",");
43 
44  if(colors.size() < 3) {
45  return false;
46  }
47 
48  c.setRed(colors[0].toInt());

50  c.setBlue(colors[2].toInt());
51 
52  if(colors.size() == 4){
53  c.setAlpha(colors[3].toInt());
54  }
55 
56  else{
57  c.setAlpha(255);

59 
60  return true;
