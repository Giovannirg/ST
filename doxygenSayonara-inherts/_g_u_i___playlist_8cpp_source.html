307  int n_rows = 0;

314

316  PlaylistConstPtr pl = _playlist->get_playlist_at(idx);
317  if(pl){
318  dur_ms = pl->get_running_time();

320 
321  if(dur_ms > 0){
322  time_str = Helper::cvt_ms_to_string(dur_ms, true, false);
323  }
324 
325  cur_view = get_current_view();
326  if(cur_view){
327  n_rows = cur_view->get_num_rows();
328  }
329 
330  playlist_string = text + QString::number(n_rows);
331 
332 
333  if(n_rows == 0){
334  playlist_string = tr("Playlist empty");
335  }
336 
337  else if(n_rows == 1) {
338  playlist_string += " " + Lang::get(Lang::Track);
339  }
340 
341  else {
342  playlist_string += " " + Lang::get(Lang::Tracks);
343  }
344 
345  if( dur_ms > 0 ){
346  playlist_string += " - " + time_str;
347  }
348 
349  ui->lab_totalTime->setText(playlist_string);
350  ui->lab_totalTime->setContentsMargins(0, 2, 0, 2);
351 }
