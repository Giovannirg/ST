81 bool CoverLookup::fetch_cover(const CoverLocation& cl, bool also_www)
82 {
83  // Look, if cover exists in .Sayonara/covers
84  if( QFile::exists(cl.cover_path()) && _m->n_covers == 1 )
85  {
86  emit sig_cover_found(cl.cover_path());
87  emit sig_finished(true);
88  return true;
89  }
90 
91  // For one cover, we also can use the local cover path
92  if(!cl.local_paths().isEmpty() && _m->n_covers == 1)
93  {
94  emit sig_cover_found(cl.local_path(0));
95  emit sig_finished(true);
96  return true;
97  }
98 
99  // we have to fetch the cover from the internet
100  if(also_www){
101  if(!start_new_thread( cl )){
102  return false;
103  }
104  }
105 
106  else{
107  return false;
108  }
109 
110  return true;
111 }
