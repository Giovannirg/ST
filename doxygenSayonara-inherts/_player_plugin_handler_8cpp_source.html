60  connect(p, &PlayerPluginInterface::sig_action_triggered, this, &PlayerPluginHandler::plugin_action_triggered);
61  connect(p, &PlayerPluginInterface::sig_reload, this, &PlayerPluginHandler::reload_plugin);

64  QString last_plugin = _settings->get(Set::Player_ShownPlugin);
