46 bool ConvertEngine::set_uri(const MetaData& md) {
47  QUrl url;
48  gchar* uri;
49  gchar* target_uri;
50  QString cvt_target_path;
51  bool success = false;
52 
53  if(_uri){
54  g_free(_uri);
55  _uri = nullptr;
56  }
57 
58  if(_target_uri){
59  g_free(_target_uri);
60  _target_uri = nullptr;
61  }
62 
63  _playing_stream = Helper::File::is_www( md.filepath() );
64 
65  if (_playing_stream) {
66  url = QUrl(md.filepath());
67  uri = url.toString().toUtf8().data();
68  }
69 
70  // no stream (not quite right because of mms, rtsp or other streams
71  // normal filepath -> no uri
72  else if (!md.filepath().contains("://")) {
73  url = QUrl::fromLocalFile(md.filepath());
74  uri = url.toString().toUtf8().data();
75  }
76 
77  else {
78  uri = md.filepath().toUtf8().data();
79  }
80 
81  QString filename = Helper::File::get_filename_of_path(md.filepath());
82  int idx = filename.lastIndexOf(".");
83  if(idx > 0) {
84  filename = filename.left(idx);
85  }
86 
87  cvt_target_path = _settings->get(Set::Engine_CovertTargetPath);
88  filename = cvt_target_path + "/" + filename + ".mp3";
89 
90  target_uri = filename.toUtf8().data();
91 
92  _uri = g_strdup(uri);
93  _target_uri = g_strdup(target_uri);
94 
95  success = _pipeline->set_uri(g_strdup(uri));
96  _pipeline->set_target_uri(target_uri);
97 
98  _md_target = md;
99  _md_target.set_filepath(filename);
100 
101  return success;
102 }
