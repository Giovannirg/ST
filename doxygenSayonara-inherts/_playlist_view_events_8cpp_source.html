145 void PlaylistView::keyPressEvent(QKeyEvent* event)
146 {
147  int key = event->key();
148 
149  if((key == Qt::Key_Up || key == Qt::Key_Down))
150  {
151  SP::Set<int> selections = get_selected_items();
152  bool ctrl_pressed = (event->modifiers() & Qt::ControlModifier);
153 
154  if( ctrl_pressed && !selections.isEmpty() )
155  {
156  SP::Set<int> new_selections;
157 
158  int min_row = *selections.begin();
159  int max_row = *selections.rbegin();
160 
161  if(key == Qt::Key_Up){
162  if(min_row == 0){
163  return;
164  }
165  _model->move_rows(selections, min_row - 1);
166  for(int i=min_row - 1; i<(int) selections.size() + min_row - 1; i++){
167  new_selections.insert(i);
168  }
169  }
170 
171  else{
172  if(max_row >= _model->rowCount() - 1){
173  return;
174  }
175  _model->move_rows(selections, max_row + 2);
176  for(int i=min_row + 1; i<(int) selections.size() + min_row + 1; i++){
177  new_selections.insert(i);
178  }
179  }
180 
181  if(new_selections.size() > 0){
182  this->select_rows(new_selections);
183  }
184  }
185 
186  else if(selections.isEmpty()) {
187  if(_model->rowCount() > 0) {
188  if(key == Qt::Key_Up){
189  select_row(_model->rowCount() - 1);
190  }
191  else{
192  select_row(0);
193  }
194  }
195  return;
196  }
197  }
198 
199  if(event->matches(QKeySequence::SelectAll)){
200  select_all();
201  return;
202  }
203 
204  else if(event->matches(QKeySequence::Delete)){
205  this->remove_cur_selected_rows();
206  return;
207  }
208 
209  SearchableListView::keyPressEvent(event);
210 
211  if(!event->isAccepted() ) {
212  return;
213  }
214 
215  int new_row = -1;
216  int min_row = get_min_selected_item();
217 
218  switch(key) {
219  case Qt::Key_End:
220  new_row = _model->rowCount() - 1;
221  break;
222 
223  case Qt::Key_Home:
224  new_row = 0;
225  break;
226 
227  case Qt::Key_Left:
228  if(event->modifiers() & Qt::ControlModifier){
229  emit sig_left_tab_clicked();
230  }
231  break;
232 
233  case Qt::Key_Right:
234  if(event->modifiers() & Qt::ControlModifier){
235  emit sig_right_tab_clicked();
236  }
237  break;
238 
239  case Qt::Key_Return:
240  case Qt::Key_Enter:
241  if(min_row >= 0){
242  _model->set_current_track(min_row);
243  emit sig_double_clicked(min_row);
244  }
245 
246  break;
247 
248  default:
249  break;
250  }
251 
252  if(new_row != -1) {
253  goto_row(new_row);
254  }
255 }
