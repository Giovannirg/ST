173 void GUI_Equalizer::sli_changed(int idx, int new_val) {
174  ui->btn_tool->show_action(ContextMenu::EntryUndo, true);
175 
176  EqSlider* s = _sliders[idx];
177  s->getLabel()->setText(calc_lab(new_val));
178 
179  _engine->set_equalizer(idx, new_val);
180 
181  // this slider has been changed actively
182  if( idx == _active_idx && ui->cb_gauss->isChecked() ){
183  int delta = new_val - _old_val[idx];
184 
185  for(int i=idx-9; i<idx+9; i++){
186  if(i < 0) continue;
187  if(i == idx) continue;
188  if(i >= _sliders.size()) break;
189 
190  // how far is the slider away from me?
191  int x = abs(_active_idx - i);
192 
193  if(x > 4) continue;
194 
195  double new_val = _old_val[i] + (delta * scale[x]);
196 
197  _sliders[i]->setValue(new_val);
198  }
199  }
200 }
