36 MetaDataInfo::MetaDataInfo(const MetaDataList& v_md) :
37  QObject(nullptr),
38  SayonaraClass()
39 {
40  if(v_md.isEmpty()) {
41  return;
42  }
43 
44  quint64 length = 0;
45  quint64 filesize = 0;
46  quint16 year_min = std::numeric_limits<quint16>::max();
47  quint16 year_max = 0;
48  quint32 bitrate_min = std::numeric_limits<quint32>::max();
49  quint32 bitrate_max = 0;
50  quint16 tracknum = 0;
51  bool calc_track_num = (v_md.size() == 1);
52 
53  QStringList genres;
54  QMap<QString, QStringList> values;
55 
56  _db = DB::getInstance(v_md[0]);
57 
58  for(const MetaData& md : v_md )
59  {
60  _artists.insert(md.artist);
61  _albums.insert(md.album);
62  _album_artists.insert(md.album_artist());
63 
64  _album_ids.insert(md.album_id);
65  _artist_ids.insert(md.artist_id);
66  _album_artist_ids.insert(md.album_artist_id());
67 
68  length += md.length_ms;
69  filesize += md.filesize;
70 
71  if(calc_track_num){
72  tracknum = md.track_num;
73  }
74 
75  // bitrate
76  if(md.bitrate != 0){
77  bitrate_min = std::min(md.bitrate, bitrate_min);
78  }
79 
80  bitrate_max = std::max(md.bitrate, bitrate_max);
81 
82  // year
83  if(md.year != 0) {
84  year_min = std::min(year_min, md.year);
85  }
86 
87  year_max = std::max(year_max, md.year);
88 
89  // custom fields
90  const QList<CustomField>& custom_fields = md.get_custom_fields();
91 
92  for(const CustomField& field : custom_fields){
93  QString name = field.get_display_name();
94  QString value = field.get_value();
95  if(value.isEmpty()){
96  continue;
97  }
98 
99  if(!values[name].contains(value)){
100  values[name] << value;
101  }
102  }
103 
104  // genre
105  genres = md.genres_to_list();
106 
107  // paths
108  if(!Helper::File::is_www(md.filepath())){
109  QString filename, dir;
110  Helper::File::split_filename(md.filepath(), dir, filename);
111  if( !_paths.contains(dir)){
112  _paths << dir;
113  }
114  }
115 
116  else{
117  _paths << md.filepath();
118  }
119  }
120 
121  for(const QString& name : values.keys()){
122  _additional_info[name] = values[name].join("<br />");
123  }
124 
125  if(bitrate_max > 0){
126  insert_interval(InfoStrings::Bitrate, bitrate_min / 1000, bitrate_max / 1000);
127  }
128 
129  if(year_max > 0){
130  insert_interval(InfoStrings::Year, year_min, year_max);
131  }
132 
133  insert_number(InfoStrings::nTracks, v_md.size());
134  insert_filesize(filesize);
135  insert_playing_time(length);
136  insert_genre(genres);
137 
138  set_header(v_md);
139  set_subheader(tracknum);
140  set_cover_location(v_md);
141 }
