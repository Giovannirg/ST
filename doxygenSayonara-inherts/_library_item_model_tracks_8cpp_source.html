55 QVariant LibraryItemModelTracks::data(const QModelIndex &index, int role) const
56 {
57  int row = index.row();
58  int col = index.column();
59 
60  if (!index.isValid())
61  return QVariant();
62 
63  if (row >= _m->tracks.size())
64  return QVariant();
65 
66  ColumnIndex::Track idx_col = (ColumnIndex::Track) col;
67 
68  if (role == Qt::TextAlignmentRole) {
69  if (idx_col == ColumnIndex::Track::TrackNumber ||
70  idx_col == ColumnIndex::Track::Bitrate ||
71  idx_col == ColumnIndex::Track::Length ||
72  idx_col == ColumnIndex::Track::Year ||
73  idx_col == ColumnIndex::Track::Filesize)
74  {
75  return Qt::AlignRight + Qt::AlignVCenter;
76  }
77 
78  else return Qt::AlignLeft + Qt::AlignVCenter;
79  }
80 
81  else if (role == Qt::DisplayRole || role==Qt::EditRole) {
82  const MetaData& md = _m->tracks.at(row);
83 
84  switch(idx_col) {
85  case ColumnIndex::Track::TrackNumber:
86  return QVariant( md.track_num );
87 
88  case ColumnIndex::Track::Title:
89  return QVariant( md.title );
90 
91  case ColumnIndex::Track::Artist:
92  return QVariant( md.artist );
93 
94  case ColumnIndex::Track::Length:
95  return QVariant( Helper::cvt_ms_to_string(md.length_ms) );
96 
97  case ColumnIndex::Track::Album:
98  return QVariant(md.album);
99 
100  case ColumnIndex::Track::Year:
101  if(md.year == 0){
102  return Lang::get(Lang::None);
103  }
104 
105  return md.year;
106 
107  case ColumnIndex::Track::Bitrate:
108  return QString::number(md.bitrate / 1000) + " kbit/s";
109 
110  case ColumnIndex::Track::Filesize:
111  return Helper::File::calc_filesize_str(md.filesize);
112 
113  case ColumnIndex::Track::Rating:
114  return QVariant(md.rating);
115  default:
116  return QVariant();
117  }
118  }
119 
120  return QVariant();
121 }
