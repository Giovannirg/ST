37 bool DatabaseLibrary::storeMetadata(const MetaDataList& v_md)
38 {
39  bool success = true;
40 
41  if(v_md.isEmpty()) {
42  return success;

44 
45  _db.transaction();
46 
47  DatabaseAlbums db_albums(_db, v_md.first().db_id);
48  DatabaseArtists db_artists(_db, v_md.first().db_id);

50 
51  AlbumList albums;

54  QHash<QString, Artist> artist_map;
55 
56  db_albums.getAllAlbums(albums, true);

58 
59  for(const Album& album : albums){

62 
63  for(const Artist& artist : artists){
64  artist_map[artist.name] = artist;
65  }
66 
67  albums.clear();
68  artists.clear();
69 
70  for(MetaData md : v_md) {
71  int artist_id, album_id, album_artist_id;
72  //first check if we know the artist and its id
73  Album album = album_map[md.album];
74  if(album.id < 0) {
75  album_id = db_albums.insertAlbumIntoDatabase(md.album);
76  album.id = album_id;
77  album_map[md.album] = album;
78  }
79 
80  else{
81  album_id = album.id;
82  }
83 
84  Artist artist = artist_map[md.artist];
85  if (artist.id < 0) {
86  artist_id = db_artists.insertArtistIntoDatabase(md.artist);
87  artist.id = artist_id;
88  artist_map[md.artist] = artist;
89  }
90 
91  else{
92  artist_id = artist.id;
93  }
94 
95  if(md.album_artist_id() == -1){
96  md.set_album_artist_id(artist_id);
97  }
98 
99  if(md.album_artist().isEmpty()){
100  md.set_album_artist(md.artist);
101  }
102 
103  Artist album_artist = artist_map[md.album_artist()];
104  if (album_artist.id < 0) {
105  if(md.album_artist().isEmpty()){
106  album_artist_id = -1;
107  }
108 
109  else{
110  album_artist_id = db_artists.insertArtistIntoDatabase(md.album_artist());
111  album_artist.id = album_artist_id;
112  artist_map[md.album_artist()] = album_artist;
113  }
114  }
115 
116  else{
117  album_artist_id = album_artist.id;
118  }
119 
120  md.album_id = album_id;
121  md.artist_id = artist_id;
122 
123  if(album_id == -1 || artist_id == -1){
124  sp_log(Log::Warning) << "Cannot insert artist or album of " << md.filepath();
125  continue;
126  }
127 
128  db_tracks.insertTrackIntoDatabase(md, artist_id, album_id, album_artist_id);
129  }
130 
131  success = _db.commit();
132 
133  return success;
134 }
