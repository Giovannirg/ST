79 void SomaFMLibrary::soma_website_fetched()
80 {
81  AsyncWebAccess* awa = static_cast<AsyncWebAccess*>(sender());
82  QList<SomaFMStation> stations;
83 
84  if(awa->status() != AsyncWebAccess::Status::GotData){
85  awa->deleteLater();
86  emit sig_stations_loaded(stations);
87  return;
88  }
89 
90  QString content = QString::fromUtf8(awa->data());
91  QStringList station_contents = content.split("<li");
92 
93 
94  for(const QString& station_content : station_contents){
95  SomaFMStation station(station_content);
96  if(!station.is_valid()){
97  continue;
98  }
99 
100  QString station_name = station.get_name();
101 
102  bool loved = _m->qsettings->value(station_name, false).toBool();
103 
104  station.set_loved( loved );
105 
106  _m->station_map[station_name] = station;
107  stations << station;
108  }
109 
110  sort_stations(stations);
111  emit sig_stations_loaded(stations);
112 
113  awa->deleteLater();
114 }

233 void SomaFMLibrary::sort_stations(QList<SomaFMStation>& stations)
234 {
235  auto lambda = [](const SomaFMStation& s1, const SomaFMStation& s2){
236  if(s1.is_loved() && !s2.is_loved()){
237  return true;

239 
240  else if(!s1.is_loved() && s2.is_loved()){
241  return false;
242  }
243 
244  return s1.get_name() < s2.get_name();
245  };
246
