141 bool PlaybackPipeline::add_and_link_elements()
142 {
143  bool success;
144  GstPadTemplate* tee_src_pad_template;
145 
146  gst_bin_add_many(GST_BIN(_pipeline),
147  _audio_src, _audio_convert, _equalizer, _tee,
148 
149  _eq_queue, _volume, _audio_sink,
150  _level_queue, _level, _level_sink,
151  _spectrum_queue, _spectrum, _spectrum_sink,
152 
153  nullptr);
154 
155  /* before tee */
156  success = gst_element_link_many(_audio_convert, _equalizer, _tee, nullptr);
157  if(!_test_and_error_bool(success, "Engine: Cannot link audio convert with tee")){
158  return false;
159  }
160 
161  /* standard output branch */
162  success = gst_element_link_many(_eq_queue, _volume, /*_speed,*/ _audio_sink, nullptr);
163  if(!_test_and_error_bool(success, "Engine: Cannot link eq with audio sink")) {
164  return false;
165  }
166 
167 
168  /* level branch */
169  success = gst_element_link_many(_level_queue, _level, _level_sink, nullptr);
170  if(!_test_and_error_bool(success, "Engine: Cannot link Level pipeline")){
171  return false;
172  }
173 
174 
175  /* spectrum branch */
176  success = gst_element_link_many(_spectrum_queue, _spectrum, _spectrum_sink, nullptr);
177 
178  if(!_test_and_error_bool(success, "Engine: Cannot link Spectrum pipeline")){
179  return false;
180  }
181 
182 
183  /* lame branch (optional) */
184  if(_lame){
185  gst_bin_add_many(GST_BIN(_pipeline), _lame_queue, _lame_converter, _lame_resampler, _lame, _lame_app_sink, nullptr);
186  success = gst_element_link_many( _lame_queue, _lame_converter, _lame_resampler, _lame, _lame_app_sink, nullptr);
187  _test_and_error_bool(success, "Engine: Cannot link lame stuff");
188  }
189 
190  /* stream rippper branch (optional) */
191  if(_file_sink){
192  gst_bin_add_many(GST_BIN(_pipeline), _file_queue, _file_converter, _file_resampler, _file_lame, _file_sink, nullptr);
193  success = gst_element_link_many( _file_queue, _file_converter, _file_resampler, _file_lame, _file_sink, nullptr);
194  _test_and_error_bool(success, "Engine: Cannot link streamripper stuff");
195  }
196 
197  /* create tee pads */
198  tee_src_pad_template = gst_element_class_get_pad_template (GST_ELEMENT_GET_CLASS (_tee), "src_%u");
199  if(!_test_and_error(tee_src_pad_template, "Engine: _tee_src_pad_template is nullptr")) {
200  return false;
201  }
202 
203  /* connect branches with tee */
204  success = tee_connect(_tee, tee_src_pad_template, _level_queue, "Level");
205  if(!_test_and_error_bool(success, "Engine: Cannot link level queue with tee")){
206  return false;
207  }
208 
209  tee_connect(_tee, tee_src_pad_template, _spectrum_queue, "Spectrum");
210  if(!_test_and_error_bool(success, "Engine: Cannot link spectrum queue with tee")){
211  return false;
212  }
213 
214  tee_connect(_tee, tee_src_pad_template, _eq_queue, "Equalizer");
215  if(!_test_and_error_bool(success, "Engine: Cannot link eq queue with tee")){
216  return false;
217  }
218 
219  if(_lame){
220  success = tee_connect(_tee, tee_src_pad_template, _lame_queue, "Lame");
221  if(!_test_and_error_bool(success, "Engine: Cannot link lame queue with tee")){
222  _settings->set(SetNoDB::MP3enc_found, false);
223  }
224  }
225 
226  if(_file_sink){
227  success = tee_connect(_tee, tee_src_pad_template, _file_queue, "Streamripper");
228  if(!_test_and_error_bool(success, "Engine: Cannot link streamripper stuff")){
229  _settings->set(Set::Engine_SR_Active, false);
230  }
231  }
232 
233  return true;
234 }

333 void PlaybackPipeline::init_equalizer()
334 {
335  int last_idx = _settings->get(Set::Eq_Last);
336  QList<EQ_Setting> presets = _settings->get(Set::Eq_List);
337  presets.push_front(EQ_Setting());
338 
339  if( !between(last_idx, presets)){
340  last_idx = 0;
341  }
342 
343  EQ_Setting last_preset = presets[last_idx];
344  QList<int> values = last_preset.values();
345 
346  int i=0;
347 
348  for(auto it=values.begin(); it != values.end(); it++){
349  double new_val;
350  if (*it > 0) {
351  new_val = *it * 0.25;
352  }
353 
354  else{
355  new_val = *it * 0.75;
356  }
357 
358  set_eq_band(QString("band") + QString::number(i), new_val);
359  i++;
360  }
361 }
