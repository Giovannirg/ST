182 void GUI_AbstractLibrary::text_line_edited(const QString &search)
183 {
184  if(search.startsWith("f:", Qt::CaseInsensitive)) {
185  _combo_search->setCurrentIndex(0);
186  _le_search->setText("");
187  }
188 
189  else if(search.startsWith("g:", Qt::CaseInsensitive)) {
190  _combo_search->setCurrentIndex(1);
191  _le_search->setText("");
192  }
193 
194  else if(search.startsWith("p:", Qt::CaseInsensitive)) {
195  _combo_search->setCurrentIndex(2);
196  _le_search->setText("");
197  }
198 
199  Library::SearchModeMask mask = _settings->get(Set::Lib_SearchMode);
200  Library::Filter filter;
201 
202  QString text = search;
203 
204  switch( _combo_search->currentIndex() )
205  {
206  case 1:
207  filter.set_mode(Library::Filter::Genre);
208  break;
209 
210  case 2:
211  filter.set_mode(Library::Filter::Filename);
212  break;
213 
214  case 0:
215  text = Library::convert_search_string(search, mask);
216  filter.set_mode(Library::Filter::Fulltext);
217  break;
218  default:
219  break;
220  }
221 
222  if(search.size() < 3){
223  filter.clear();
224  }
225 
226  else{
227  filter.set_filtertext( QString("%") + text + QString("%") );
228  }
229 
230  _cur_searchfilter = filter;
231 
232  _library->psl_filter_changed(filter);
233 }
