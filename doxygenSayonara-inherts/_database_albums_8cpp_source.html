34 QString DatabaseAlbums::fetch_query_albums(bool also_empty) const
35 {
36  QString sql=
37  "SELECT "
38  " albums.albumID AS albumID"
39  ", albums.name AS albumName"
40  ", SUM(tracks.length) / 1000 AS albumLength"
41  ", albums.rating AS albumRating"
42  ", COUNT(DISTINCT tracks.trackid) AS trackCount"
43  ", MAX(tracks.year) AS albumYear"
44  ", GROUP_CONCAT(DISTINCT artists.name)"
45  ", GROUP_CONCAT(DISTINCT tracks.discnumber)"
46  ", GROUP_CONCAT(DISTINCT albumArtists.name)"
47  " FROM albums ";
48 
49  QString join = " INNER JOIN ";
50  if(also_empty){
51  join = " LEFT OUTER JOIN ";
52  }
53 
54  sql += join + " tracks ON tracks.albumID = albums.albumID " +
55  join + " artists ON ("
56  " tracks.artistID = artists.artistID OR "
57  " tracks.albumArtistID = artists.artistID"
58  " ) " +
59  " LEFT OUTER JOIN artists albumArtists ON "
60  " tracks.albumArtistID = albumArtists.artistID ";
61 
62  return sql;
63 }

66 bool DatabaseAlbums::db_fetch_albums(SayonaraQuery& q, AlbumList& result)
67 {
68  result.clear();
69 
70  if (!q.exec()) {
71  q.show_error("Could not get all albums from database");
72  return false;
73  }
74 
75  if(!q.last()){
76  return true;
77  }
78 
79  for(bool is_element=q.first(); is_element; is_element = q.next())
80  {
81  Album album;
82 
83  album.id = q.value(0).toInt();
84  album.name = q.value(1).toString().trimmed();
85  album.length_sec = q.value(2).toInt();
86  album.rating = q.value(3).toInt();
87  album.num_songs = q.value(4).toInt();
88  album.year = q.value(5).toInt();
89  album.artists = q.value(6).toString().split(',');
90  album.set_album_artists(q.value(8).toString().split(','));
91 
92  album.discnumbers.clear();
93  QStringList discs = q.value(7).toString().split(',');
94  discs.removeDuplicates();
95 
96  for(const QString& disc : discs) {
97  album.discnumbers << disc.toInt();
98  }
99 
100  if(album.discnumbers.isEmpty()) {
101  album.discnumbers << 1;
102  }
103 
104  album.n_discs = album.discnumbers.size();
105  album.is_sampler = (album.artists.size() > 1);
106  album.db_id = _module_db_id;
107 
108  result << album;
109  };
110 
111  return true;
112 }
