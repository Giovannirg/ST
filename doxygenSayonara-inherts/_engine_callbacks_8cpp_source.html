49 static bool parse_image(GstTagList* tags, QImage& img)
50 {
51  GstSample* sample;
52 
53  bool success = gst_tag_list_get_sample(tags, GST_TAG_IMAGE, &sample);
54  if(!success){
55  success = gst_tag_list_get_sample(tags, GST_TAG_PREVIEW_IMAGE, &sample);
56  if(!success){
57  return false;
58  }
59  }
60 
61  GstCaps* caps = gst_sample_get_caps(sample);
62  if(!caps){
63  return false;
64  }
65 
66  gchar* mimetype = gst_caps_to_string(caps);
67  if(mimetype == nullptr){
68  return false;
69  }
70 
71  QString mime(mimetype);
72  g_free(mimetype); mimetype = nullptr;
73  //sp_log(Log::Debug, this) << "Mime type: " << mimetype;
74  QRegExp re(".*(image/[a-z|A-Z]+).*");
75  if(re.indexIn(mime) >= 0){
76  mime = re.cap(1);
77  }
78 
79  GstBuffer* buffer = gst_sample_get_buffer( sample );
80  if(!buffer){
81  gst_sample_unref(sample);
82  return false;
83  }
84 
85  gsize size = gst_buffer_get_size(buffer);
86  if(size == 0){
87  gst_sample_unref(sample);
88  return false;
89  }
90 
91  gchar* data = new gchar[size];
92  size = gst_buffer_extract(buffer, 0, data, size);
93 
94  if(size == 0){
95  delete[] data;
96 
97  gst_sample_unref(sample);
98 
99  return false;
100  }
101 
102  img = QImage::fromData((const uchar*) data, size, mime.toLocal8Bit().data());
103 
104  delete[] data;
105  gst_sample_unref(sample);
106 
107  return (!img.isNull());
108 }

112 gboolean EngineCallbacks::bus_state_changed(GstBus* bus, GstMessage* msg, gpointer data)
113 {
114  Q_UNUSED(bus);
115 
116  Engine* engine = static_cast<Engine*>(data);
117  if(!engine){
118  return true;
119  }
120 
121  GstMessageType msg_type = GST_MESSAGE_TYPE(msg);
122  QString msg_src_name = QString(GST_MESSAGE_SRC_NAME(msg)).toLower();
123  GstElement* src = reinterpret_cast<GstElement*>(msg->src);
124 
125  switch (msg_type)
126  {
127  case GST_MESSAGE_EOS:
128 
129  if( !msg_src_name.contains("sr_filesink") &&
130  !msg_src_name.contains("level_sink") &&
131  !msg_src_name.contains("spectrum_sink") &&
132  !msg_src_name.contains("pipeline"))
133  {
134  sp_log(Log::Debug) << "EOF reached: " << msg_src_name;
135  break;
136  }
137 
138  engine->set_track_finished(src);
139 
140  break;
141 
142  case GST_MESSAGE_ELEMENT:
143 
144  if(msg_src_name.compare("spectrum") == 0){
145  return spectrum_handler(bus, msg, engine);
146  }
147 
148  if(msg_src_name.compare("level") == 0){
149  return level_handler(bus, msg, engine);
150  }
151 
152  break;
153 
154  case GST_MESSAGE_SEGMENT_DONE:
155  sp_log(Log::Debug) << "Segment done: " << msg_src_name;
156  break;
157 
158  case GST_MESSAGE_TAG:
159  {
160  GstTagList* tags;
161  gchar* title;
162  QImage img;
163  bool success;
164  quint32 bitrate;
165 
166  if( msg_src_name.compare("sr_filesink") == 0 ||
167  msg_src_name.compare("level_sink") == 0 ||
168  msg_src_name.compare("spectrum_sink") == 0)
169  {
170  break;
171  }
172 
173  tags = nullptr;
174  gst_message_parse_tag(msg, &tags);
175 
176  if(!tags){
177  break;
178  }
179 
180  success = parse_image(tags, img);
181  if(success){
182  engine->update_cover(img, src);
183  }
184 
185  success = gst_tag_list_get_uint(tags, GST_TAG_BITRATE, &bitrate);
186  if(success){
187  engine->update_bitrate((bitrate / 1000) * 1000, src);
188  }
189 
190  success = gst_tag_list_get_string(tags, GST_TAG_TITLE, (gchar**) &title);
191  if(success){
192  MetaData md;
193  md.title = title;
194  g_free(title);
195  engine->update_md(md, src);
196  }
197 
198  gst_tag_list_unref(tags);
199  }
200 
201  break;
202 
203  case GST_MESSAGE_STATE_CHANGED:
204  GstState old_state, new_state, pending_state;
205 
206  gst_message_parse_state_changed(msg, &old_state, &new_state, &pending_state);
207  /*sp_log(Log::Debug, this) << GST_MESSAGE_SRC_NAME(msg) << ": "
208  << "State changed from "
209  << gst_element_state_get_name(old_state)
210  << " to "
211  << gst_element_state_get_name(new_state)
212  << " pending: "
213  << gst_element_state_get_name(pending_state);*/
214 
215  if(!msg_src_name.contains("pipeline", Qt::CaseInsensitive)){
216  break;
217  }
218 
219  if( new_state == GST_STATE_PLAYING ||
220  new_state == GST_STATE_PAUSED)
221  {
222  engine->set_track_ready(src);
223  }
224 
225  break;
226 
227  case GST_MESSAGE_BUFFERING:
228 
229  gint percent;
230 
231  gint avg_in, avg_out;
232  gint64 buffering_left;
233 
234  GstBufferingMode mode;
235  gst_message_parse_buffering(msg, &percent);
236  gst_message_parse_buffering_stats(msg, &mode, &avg_in, &avg_out, &buffering_left );
237 
238  /*sp_log(Log::Debug, this) << "Buffering " << percent;
239  sp_log(Log::Debug, this) << "Buffering State: " << (int) mode << " avg in: " << avg_in << " avg out: " << avg_out << " todo: " << buffering_left;*/
240 
241  engine->set_buffer_state(percent, src);
242  break;
243 
244  case GST_MESSAGE_DURATION_CHANGED:
245  engine->update_duration(src);
246  break;
247 
248  case GST_MESSAGE_INFO:
249  /*gst_message_parse_info(msg, &err, nullptr);*/
250  break;
251 
252  case GST_MESSAGE_WARNING:
253  {
254  GError* err;
255  gst_message_parse_warning(msg, &err, nullptr);
256  sp_log(Log::Warning) << "Engine " << (int) engine->get_name() << ": GST_MESSAGE_WARNING: " << err->message << ": "
257  << GST_MESSAGE_SRC_NAME(msg);
258  }
259  break;
260 
261  case GST_MESSAGE_ERROR:
262  {
263  GError* err;
264  gst_message_parse_error(msg, &err, nullptr);
265 
266  sp_log(Log::Error) << "Engine " << (int) engine->get_name() << ": GST_MESSAGE_ERROR: " << err->message << ": "
267  << GST_MESSAGE_SRC_NAME(msg);
268 
269  engine->set_track_finished(src);
270  engine->stop();
271  g_error_free(err);
272  }
273  break;
274 
275  case GST_MESSAGE_STREAM_STATUS:
276  /*GstStreamStatusType type;
277  gst_message_parse_stream_status(msg, &type, NULL);
278  sp_log(Log::Debug, this) << "Get stream status " << type;*/
279  break;
280 
281  default:
282  break;
283  }
284 
285  return true;
286 }

291 EngineCallbacks::level_handler(GstBus * bus, GstMessage * message, gpointer data)
292 {
293  Q_UNUSED(bus);
294 
295  double channel_values[2];
296 
297  PlaybackEngine* engine;
298  GValueArray* rms_arr;
299  const GstStructure* structure;
300  const gchar* name;
301  const GValue* peak_value;
302  guint n_peak_elements;
303 
304  engine = static_cast<PlaybackEngine*>(data);
305  if(!engine) {
306  return true;
307  }
308 
309  structure = gst_message_get_structure(message);
310  if(!structure) {
311  sp_log(Log::Warning) << "structure is null";
312  return true;
313  }
314 
315  name = gst_structure_get_name(structure);
316  if ( strcmp(name, "level") != 0 ) {
317  return true;
318  }
319 
320  peak_value = gst_structure_get_value(structure, "peak");
321  if(!peak_value) {
322  return true;
323  }
324 
325  rms_arr = static_cast<GValueArray*>(g_value_get_boxed(peak_value));
326 
327  n_peak_elements = rms_arr->n_values;
328  if(n_peak_elements == 0) {
329  return true;
330  }
331 
332  n_peak_elements = std::min((guint) 2, n_peak_elements);
333  for(guint i=0; i<n_peak_elements; i++) {
334  double d;
335  const GValue* val;
336 
337  val = rms_arr->values + i;
338 
339  if(!G_VALUE_HOLDS_DOUBLE(val)) {
340  sp_log(Log::Debug) << "Could not find a double";
341  break;
342  }
343 
344  d = g_value_get_double(val);
345  if(d < 0){
346  channel_values[i] = d;
347  }
348  }
349 
350  if(n_peak_elements >= 2) {
351  engine->set_level(channel_values[0], channel_values[1]);
352  }
353 
354  else if(n_peak_elements == 1) {
355  engine->set_level(channel_values[0], channel_values[0]);
356  }
357 
358  return true;
359 }
