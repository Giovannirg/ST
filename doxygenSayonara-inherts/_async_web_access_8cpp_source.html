88 void AsyncWebAccess::run(const QString& url, int timeout)
89 {
90  _m->status = AsyncWebAccess::Status::NoData;
91  _m->header.clear();
92  _m->data.clear();
93  _m->url = url;
94  _m->nam->clearAccessCache();
95  _m->ignore_finished = false;
96 
97  QRegExp re("(itpc|feed)://");
98  if(re.indexIn(url) >= 0){
99  _m->url.replace(re, "http://");
100  }
101 
102  QNetworkRequest request;
103  request.setUrl(_m->url);
104 
105  QString user_agent;
106 
107  switch(_m->behavior){
108  case AsyncWebAccess::Behavior::AsSayonara:
109  user_agent = "Sayonara/" + QString(SAYONARA_VERSION) ;
110  break;
111 
112  case AsyncWebAccess::Behavior::AsBrowser:
113  user_agent = "Mozilla/5.0 (Linux; rv:35.0) Gecko/20100101 Firefox/35.0";
114  break;
115 
116  case AsyncWebAccess::Behavior::Random:
117  user_agent = Helper::get_random_string(Helper::get_random_number(8, 16));
118  break;
119  case AsyncWebAccess::Behavior::None:
120  default:
121  break;
122  }
123 
124  request.setHeader(QNetworkRequest::UserAgentHeader, user_agent);
125 
126  sp_log(Log::Debug, this) << "Call " << request.url().toString();
127  _m->reply = _m->nam->get(request);
128 
129  connect(_m->reply, &QNetworkReply::readyRead, this, &AsyncWebAccess::data_available);
130  connect(_m->reply, &QNetworkReply::finished, this, &AsyncWebAccess::finished);
131 
132  if(timeout > 0){
133  _m->timer->start(timeout);
134  }
135 }

189 void AsyncWebAccess::finished()
190 {
191  QNetworkReply* reply = static_cast<QNetworkReply*>(sender());
192 
193  if(_m->ignore_finished){
194  _m->abort_request();
195  _m->delete_reply();
196  return;
197  }
198 
199  QNetworkReply::NetworkError err = reply->error();
200  bool success = (err == QNetworkReply::NoError);
201 
202  sp_log(Log::Develop, this) << reply->url().toString() << " finished: " << (int) err;
203  if(success)
204  {
205  QString redirect_url = reply->attribute(QNetworkRequest::RedirectionTargetAttribute).toString();
206  if( !redirect_url.isEmpty() &&
207  redirect_url != _m->url)
208  {
209  redirect_request(redirect_url);
210  return;
211  }
212 
213  if( reply->bytesAvailable() > 0 )
214  {
215  sp_log(Log::Develop, this) << "Got " << reply->bytesAvailable() << " bytes";
216  _m->data = reply->readAll();
217  _m->status = AsyncWebAccess::Status::GotData;
218  }
219 
220  else {
221  sp_log(Log::Develop, this) << "Answer contains no data";
222  _m->status = AsyncWebAccess::Status::NoData;
223  _m->data.clear();
224  }
225  }
226 
227  else {
228 
229  sp_log(Log::Warning, this) << "Cannot open " << _m->url << ": "
230  << reply->errorString()
231  << " (" << (int) err << ")";
232 
233  if(err == QNetworkReply::TimeoutError)
234  {
235  _m->status = AsyncWebAccess::Status::Timeout;
236  }
237 
238  else if(err == QNetworkReply::RemoteHostClosedError)
239  {
240  _m->status = AsyncWebAccess::Status::NoHttp;
241  }
242 
243  else {
244  _m->status = AsyncWebAccess::Status::Error;
245  }
246  }
247 
248  _m->abort_request();
249  _m->delete_reply();
250 
251  emit sig_finished();
252 }
