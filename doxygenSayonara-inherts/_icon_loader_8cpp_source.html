75 QStringList IconLoader::load_ancestors(const QString &index_theme_file)
76 {
77  QFile f(index_theme_file);
78  if(!f.open(QFile::ReadOnly)){
79  return QStringList();
80  }
81 
82  QRegExp re_inherit(".*Inhertits\\d=\\d(.*)");
83 
84  QStringList ancestor_list;
85 
86  while(f.canReadLine()){
87  QByteArray arr = f.readLine();
88  QString line = QString::fromLocal8Bit(arr);
89  QString ancestors;
90 
91  int idx = re_inherit.indexIn(line);
92  if(idx < 0){
93  continue;
94  }
95 
96  ancestors = re_inherit.cap(1);
97  ancestor_list = ancestors.split(',');
98  break;
99  }
100 
101  f.close();
102 
103  for(auto it=ancestor_list.begin(); it != ancestor_list.end(); it++){
104  *it = it->trimmed();
105  }
106 
107  return ancestor_list;
108 }
