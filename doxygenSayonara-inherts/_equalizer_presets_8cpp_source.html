40 EQ_Setting EQ_Setting::fromString(const QString& str)
41 {
42  EQ_Setting eq;
43  QStringList list = str.split(':');
44  if(list.size() < 11) {
45  sp_log(Log::Debug) << "no valid eq string: " << str;
46  return eq;
47  }
48 
49  eq.set_name(list.first());
50  list.pop_front();
51 
52  for(int i=0; i<list.size(); i++){
53  if( i == eq.values().size() ){
54  break;
55  }
56 
57  eq.set_value(i, list[i].toInt());
58  }
59 
60  return eq;
61 }
