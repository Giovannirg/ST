94 void AlbumCoverModel::set_max_columns(int columns)
95 {

104  _m->columns = columns;
105 
106  if(columns > old_columns) {
107  beginInsertColumns(QModelIndex(), 0, columns - old_columns - 1);

109  }
110 
111  else if(columns < old_columns) {
112 
113  beginRemoveColumns(QModelIndex(), 0, old_columns - columns - 1);
114  endRemoveColumns();
115  }
116 
117  if(rows > old_rows) {
118  beginInsertRows(QModelIndex(), 0, rows - old_rows - 1);
119  endInsertRows();
120  }
121 
122  else if(rows < old_rows) {
123  beginRemoveRows(QModelIndex(), 0, old_rows - rows - 1);
124  endRemoveRows();
125  }
126 
127  emit dataChanged(index(0, 0), index(rowCount() - 1, columnCount() - 1));
128 }

130 QVariant AlbumCoverModel::data(const QModelIndex& index, int role) const
131 {
132  if(!index.isValid()) {
133  return QVariant();
134  }
135 
136  int row = index.row();
137  int col = index.column();
138 
139  int n_columns = columnCount();
140  int lin_idx = (row * n_columns) + col;
141  if(lin_idx >= _m->albums.size()){
142  return QVariant();
143  }
144 
145  switch(role)
146  {
147  case Qt::DisplayRole:
148  return _m->albums[lin_idx].name;
149 
150  case Qt::TextAlignmentRole:
151  return Qt::AlignHCenter;
152 
153  case Qt::DecorationRole:
154  {
155  const Album& album = _m->albums[lin_idx];
156  QString hash = get_hash(album);
157  QPixmap p;
158  if(!_m->pixmaps.contains(hash)) {
159  CoverLocation cl;
160  if(!_m->cover_locations.contains(hash)){
161  cl = CoverLocation::get_cover_location(album);
162  _m->cover_locations[hash] = cl;
163  }
164 
165  else{
166  cl = _m->cover_locations[hash];
167  }
168 
169  p = QPixmap(cl.preferred_path());
170  if(!CoverLocation::isInvalidLocation(cl.preferred_path())){
171  _m->pixmaps[hash] = p;
172  }
173 
174  else {
175  if(!_m->cover_thread->isRunning()){
176  _m->cover_thread->start();
177  }
178 
179  _m->indexes[hash] = index;
180  /*sp_log(Log::Debug, this) << "";
181  sp_log(Log::Debug, this) << " **** " << "Fetch cover for " << album.name;
182  sp_log(Log::Debug, this) << "";*/
183  _m->cover_thread->add_data(hash, cl);
184  }
185  }
186 
187  else{
188  p = _m->pixmaps[hash];
189  }
190 
191  return p.scaled(_m->size, _m->size, Qt::KeepAspectRatio);;
192  }
193 
194  case Qt::SizeHintRole:
195  return get_item_size();
196 
197  default:
198  return QVariant();
199  }
200 
201  return QVariant();
202 }
