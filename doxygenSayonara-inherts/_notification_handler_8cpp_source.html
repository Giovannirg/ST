43 void NotificationHandler::register_notificator(NotificationInterface* notificator)
44 {
45  _notificators << notificator;
46 
47  QString preferred = Settings::getInstance()->get(Set::Notification_Name);
48 
49  auto lambda = [preferred](NotificationInterface* n){
50  return (n->get_name().compare(preferred, Qt::CaseInsensitive) == 0);
51  };

53  auto it = std::find_if(_notificators.begin(), _notificators.end(), lambda);
54  _cur_idx = (it - _notificators.begin());
