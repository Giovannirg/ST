190 bool PlaybackEngine::set_uri(const QString& filepath)
191 {
192  bool success = false;
193  QUrl url;
194 
195  if(_uri){
196  g_free(_uri);
197  _uri = nullptr;
198  }
199 
200  _playing_stream = Helper::File::is_www(filepath);
201 
202  // stream, but don't want to record
203  // stream is already uri
204  if (_playing_stream) {
205  url = QUrl(filepath);
206  _uri = g_strdup(url.toString().toUtf8().data());
207  }
208 
209  // no stream (not quite right because of mms, rtsp or other streams
210  // normal filepath -> no uri
211  else if (!filepath.contains("://")) {
212  url = QUrl::fromLocalFile(filepath);
213  _uri = g_strdup(url.url().toUtf8().data());
214  }
215 
216  else {
217  _uri = g_strdup(filepath.toUtf8().data());
218  }
219 
220  if(g_utf8_strlen(_uri, 1024) == 0){
221  sp_log(Log::Warning) << "uri = 0";
222  return false;
223  }
224 
225  success = _pipeline->set_uri(_uri);
226 
227  return success;
228 }
