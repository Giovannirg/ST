54 void SayonaraSelectionView::select_rows(const SP::Set<int>& indexes, int min_col, int max_col)
55 {
56  QItemSelectionModel* sel_model = this->get_selection_model();
57  if(!sel_model){
58  return;
59  }
60 
61  if(indexes.size() > 0) {
62  int first_index = indexes.first();
63  this->set_current_index(first_index);
64  }
65 
66  min_col = std::max(0, min_col);
67  min_col = std::min(min_col, get_column_count() - 1);
68  max_col = std::max(0, max_col);
69  max_col = std::min(max_col, get_column_count() - 1);
70 
71  QItemSelection sel;
72  if(indexes.size() == 1)
73  {
74  QModelIndex first_idx = get_index(indexes.first(), 0);
75  QModelIndex last_idx = get_index(indexes.first(), get_column_count() - 1);
76 
77  sel.select(first_idx, last_idx);
78  sel_model->select(sel, QItemSelectionModel::ClearAndSelect);
79  return;
80  }
81 
82 
83  // the goal is: find consecutive ranges.
84  // For every select or merge an overlap is
85  // tested, that will last really long when
86  // there are already around 500 items.
87  // probably, the runtime is O(n^2)
88 
89  // the list is pre-sorted (see the fill-function)
90 
91  // we start at the very beginning (i)
92  // let j run until the first element that is not
93  // consecutive.
94 
95  // our range is from i to the last known j.
96  // count down the j by one. So the worst that can
97  // happen, is that j is as big as i again.
98  // i is increased in the next loop, so progress is
99  // guaranteed
100 
101 
102  for(auto it=indexes.begin(); it!=indexes.end(); it++)
103  {
104  auto other_it=it;
105  auto other_predecessor=it;
106 
107  do
108  {
109  other_predecessor = other_it;
110  other_it++;
111 
112  if(other_it == indexes.end()){
113  break;
114  }
115 
116  } while(*other_it - 1 == *other_predecessor);
117 
118  // select the range
119 
120  QModelIndex min_idx = get_index(*it, min_col);
121  QModelIndex max_idx = get_index(*other_predecessor, max_col);
122  sel.select(min_idx, max_idx);
123 
124  it = other_it;
125 
126  if(it == indexes.end()){
127  break;
128  }
129  }
130 
131  sel_model->select(sel, QItemSelectionModel::ClearAndSelect);
132 }
