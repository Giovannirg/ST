40 QDomNode find_artist_node(const QDomNode& node, const QString& prefix)
41 {
42  if(node.nodeName().compare("artist", Qt::CaseInsensitive) == 0 ||
43  node.nodeName().compare("album", Qt::CaseInsensitive) == 0){
44  return node;
45  }
46 
47  if(node.hasChildNodes()){
48  return find_artist_node(node.firstChild(), prefix + " ");

50 
51  else if(!node.nextSibling().isNull()){
52  return find_artist_node(node.nextSibling(), prefix);

54

61 QStringList LFMCoverFetcher::calc_addresses_from_website(const QByteArray& website) const
62 {
63  QDomDocument doc("LastFM Cover");
64  doc.setContent(website);
65 
66  QDomNode root_node = doc.firstChild();
67  QDomNode artist_node = find_artist_node(root_node, "");
68 
69  if(artist_node.isNull()){
70  return QStringList();
71  }
72 
73  QDomNodeList nodes = artist_node.childNodes();
74  if(nodes.isEmpty()){
75  return QStringList();
76  }
77 
78  QStringList attributes; attributes << "mega" << "extralarge" << "large" << "";
79  QMap<QString, QString> lfm_covers;
80 
81  for(int i=0; i<nodes.size(); i++){
82  QDomNode node = nodes.item(i);
83  QString name = node.toElement().tagName();
84  if(name.compare("image", Qt::CaseInsensitive) == 0){
85  QDomNode attr_node = node.attributes().namedItem("size");
86  QString size_attr = attr_node.nodeValue();
87 
88  QString url = node.toElement().text();
89 
90  lfm_covers[size_attr] = url;
91  }
92  }
93 
94  QStringList ret;
95  for(const QString& attr : attributes){
96  QString url = lfm_covers[attr];
97  if(!url.isEmpty()){
98  ret << url;
99  }
100  }
101 
102  sp_log(Log::Debug, this) << "Got " << ret.size() << " addresses";
103  return ret;
104 }
