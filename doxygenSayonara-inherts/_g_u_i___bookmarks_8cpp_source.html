109  ui->cb_bookmarks->clear();

112  }
113 
114  MetaData md = _bookmarks->get_cur_track();
115 
116  ui->btn_tool->show_action(ContextMenu::EntryNew, (md.id >= 0) );
117  ui->btn_tool->show_action(ContextMenu::EntryDelete, !bookmarks.isEmpty() );
118 
119  if(md.id >= 0 && bookmarks.size() > 0){
120  ui->stackedWidget->setCurrentIndex(0);
121  }
122 
123  else{
124  ui->stackedWidget->setCurrentIndex(1);
125  }
