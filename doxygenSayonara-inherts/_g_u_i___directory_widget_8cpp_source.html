279 void GUI_DirectoryWidget::search_button_clicked()
280 {
281  if(ui->le_search->text().isEmpty()){
282  return;
283  }
284 
285  if(_m->search_term == ui->le_search->text()){
286  _m->found_idx = _m->dir_model->getNextRowIndexOf(_m->search_term, 0, QModelIndex());
287  }
288  else{
289  _m->search_term = ui->le_search->text();
290  _m->found_idx = _m->dir_model->getFirstRowIndexOf(_m->search_term);
291  ui->btn_search->setText(Lang::get(Lang::SearchNext));
292  }
293 
294  if(!_m->found_idx.isValid()){
295  return;
296  }
297 
298  if(_m->dir_model->canFetchMore(_m->found_idx)){
299  _m->dir_model->fetchMore(_m->found_idx);
300  }
301 
302  ui->tv_dirs->scrollTo(_m->found_idx, QAbstractItemView::PositionAtCenter);
303  ui->tv_dirs->selectionModel()->select(_m->found_idx, QItemSelectionModel::ClearAndSelect);
304  dir_clicked(_m->found_idx);
305 }
