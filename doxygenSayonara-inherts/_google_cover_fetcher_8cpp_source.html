36 QStringList GoogleCoverFetcher::calc_addresses_from_website(const QByteArray& website) const
37 {
38  QString regex = "(https://encrypted-tbn.+)\"";
39  QStringList addresses;
40 
41  if (website.isEmpty()) {
42  sp_log(Log::Error, this) << "Website empty";
43  return addresses;
44  }
45 
46  int idx=500;
47 
48  QString website_str = QString::fromLocal8Bit(website);
49 
50  while(true)
51  {
52  QRegExp re(regex);
53  re.setMinimal(true);
54  idx = re.indexIn(website_str, idx);
55 
56  if(idx == -1) {
57  break;
58  }
59 
60  QString str = re.cap(0);
61 
62  idx += str.length();
63  str.remove("\"");
64  addresses << str;
65  }
66 
67  sp_log(Log::Debug, this) << "Got " << addresses.size() << " Addresses";
68 
69  return addresses;
70 }

78 QString GoogleCoverFetcher::get_album_address(const QString& artist, const QString& album) const
79 {
80  QString new_album, searchstring;
81  QRegExp regex;
82 
83 
84  if(searchstring.compare("various", Qt::CaseInsensitive) != 0){
85  searchstring = QUrl::toPercentEncoding(artist);
86  }
87 
88  new_album = album;
89 
90  regex = QRegExp(QString("(\\s)?-?(\\s)?((cd)|(CD)|((d|D)((is)|(IS))(c|C|k|K)))(\\d|(\\s\\d))"));
91 
92  new_album = new_album.toLower();
93  new_album = new_album.remove(regex);
94  new_album = new_album.replace("()", "");
95  new_album = new_album.replace("( )", "");
96  new_album = new_album.trimmed();
97  new_album = QUrl::toPercentEncoding(album);
98 
99  if(searchstring.size() > 0) {
100  searchstring += "+";
101  }
102 
103  searchstring += new_album;
104 
105  return get_search_address(searchstring);
