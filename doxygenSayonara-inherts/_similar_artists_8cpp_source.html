32 static QString get_filename(const QString& artist)
33 {

63 SimilarArtists::get_similar_artists(const QString& artist) const
64 {
65  QMap<QString, double> sim_artist_map;
66  QString filename = get_filename(artist);
67  if(filename.isEmpty()){
68  return sim_artist_map;
69  }
70 
71  QByteArray content, decomp;
72  bool success = Helper::File::read_file_into_byte_arr(filename, content);
73  if(!success){
74  return sim_artist_map;
75  }
76 
77  decomp = Compressor::decompress(content);
78  if(decomp.isEmpty()){
79  return sim_artist_map;
80  }
81 
82  QStringList sim_artists = QString::fromLocal8Bit(decomp).split("\n");
83  for(const QString& sim_artist : sim_artists){
84  QStringList lst = sim_artist.split('\t');
85  if(lst.size() < 3){
86  continue;
87  }
88 
89  QString match = lst[0];
90  QString artist_name = lst[2];
91 
92  sim_artist_map[artist_name] = match.toDouble();
93  }
94 
95  return sim_artist_map;
96 }
