102 QString Library::convert_search_string(const QString& str, Library::SearchModeMask mode, const QList<QChar>& ignored_chars)
103 {
104  if(diacritic_chars.isEmpty()){
105  init_diacritic_chars();
106  }
107 
108  QString ret = str;
109  if(mode & Library::CaseInsensitve)
110  {
111  ret = str.toLower();
112  }
113 
114  if(mode & Library::NoSpecialChars)
115  {
116  QString special_chars =
117  QString::fromUtf8("\\.|'|\"|&|!|\\$|\\+|\\*|\\s|/|\\(|\\)|\\=|-|_|;|:|,|\\?|<|>|\\[|\\]|\\{|\\}|@|€");
118 
119  for(const QChar& c : ignored_chars){
120  special_chars.remove(c);
121  }
122 
123  ret.remove(QRegExp(special_chars));
124  }
125 
126  if(mode & Library::NoDiacriticChars)
127  {
128  QString cleaned_string;
129 
130  for (int i = 0; i < ret.length(); i++)
131  {
132  QString c = QString(ret[i]);
133  QString replacement;
134 
135  if(diacritic_chars.contains(c)){
136  replacement = diacritic_chars[c];
137  }
138 
139  else{
140  replacement = c;
141  }
142 
143  if(mode & Library::CaseInsensitve)
144  {
145  replacement = replacement.toLower();
146  }
147 
148  cleaned_string.append(replacement);
149  }
150 
151  ret = cleaned_string;
152  }
153 
154 // sp_log(Log::Debug, this) << "Cleaned string: " << ret;
155 
156  return ret;
157 }
