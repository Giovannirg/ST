147 QStringList DirectoryReader::find_files_rec(QDir dir, const QString& filename)
148 {
149  if(dir.canonicalPath().isEmpty()){

151  }
152 
153  if(filename.isEmpty()){

156 
157  QStringList ret;

160 
161  for(const QString& d : dirs){

163  continue;
164  }
165 
166  QFileInfo fi(d);
167  if(!fi.isDir()){
168  continue;
169  }
170 
171  dir.cd(d);
172  ret += find_files_rec(dir, filename);
173  dir.cdUp();
174  }
175 
176  for(const QString& file : files){
177  QFileInfo fi(file);
178  if(!fi.isFile()){
179  continue;
180  }
181 
182  if(file.contains(filename)){
183  ret += dir.absoluteFilePath(file);
184  }
185  }
186 
187  return ret;
