81 bool PlaylistDBWrapper::get_playlists(CustomPlaylists& playlists, DatabasePlaylist::PlaylistChooserType type, Playlist::SortOrder so)
82 {
83  Q_UNUSED(type)
84 
85  bool success;
86  CustomPlaylistSkeletons skeletons;
87 
88  success = get_all_skeletons(skeletons, so);
89  if(!success){
90  return false;
91  }
92 
93  bool load_temporary = (type == DatabasePlaylist::PlaylistChooserType::OnlyTemporary ||
94  type == DatabasePlaylist::PlaylistChooserType::TemporaryAndPermanent);
95 
96  bool load_permanent = (type == DatabasePlaylist::PlaylistChooserType::OnlyPermanent ||
97  type == DatabasePlaylist::PlaylistChooserType::TemporaryAndPermanent);
98 
99  for(const CustomPlaylistSkeleton& skeleton : skeletons){
100  CustomPlaylist pl(skeleton);
101  if(pl.id() < 0){
102  continue;
103  }
104 
105  success = _db->getPlaylistById(pl);
106 
107  if(!success){
108  continue;
109  }
110 
111  apply_tags(pl);
112 
113  if( (pl.temporary() && load_temporary) ||
114  (!pl.temporary() && load_permanent) )
115  {
116  playlists.push_back(pl);
117  }
118  }
119 
120  return true;
121 }
