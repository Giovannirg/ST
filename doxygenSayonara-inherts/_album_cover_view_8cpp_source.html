101 void AlbumCoverView::change_zoom(int zoom)
102 {
103  if(zoom != -1){
104  zoom = std::min(zoom, 200);
105  zoom = std::max(zoom, 50);
106 
107  if( zoom == _m->zoom ){
108  return;
109  }
110  }
111 
112  else{
113  zoom = _m->zoom;
114  }
115 
116  _m->zoom = zoom;
117  _m->model->set_zoom(_m->zoom);
118  _settings->set(Set::Lib_CoverZoom, _m->zoom);
119

121  int n_cols_target = (this->width() + (col_width / 10)) / col_width;
122 
123  _m->model->set_max_columns(n_cols_target);
