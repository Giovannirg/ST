66  void parse_urls()
67  {
68  QString mp3_pattern("<nobr>\\s*MP3:\\s*<a\\s+href=\"(.*)\"");
69  QString aac_pattern("<nobr>\\s*AAC:\\s*<a\\s+href=\"(.*)\"");
70  QRegExp re_mp3(mp3_pattern);
71  QRegExp re_aac(aac_pattern);
72 
73  re_mp3.setMinimal(true);
74  re_aac.setMinimal(true);
75 
76  int idx=-1;
77  do{
78  idx = re_mp3.indexIn(content, idx+1);
79  if(idx > 0){
80  QString url = complete_url(re_mp3.cap(1));
81  urls[url] = SomaFMStation::UrlType::MP3;
82  }
83  } while(idx > 0);
84 
85  idx=-1;
86  do{
87  idx = re_aac.indexIn(content, idx+1);
88 
89  if(idx > 0){
90  QString url = complete_url(re_aac.cap(1));
91  urls[url] = SomaFMStation::UrlType::AAC;
92  }
93 
94  } while(idx > 0);
95  }
