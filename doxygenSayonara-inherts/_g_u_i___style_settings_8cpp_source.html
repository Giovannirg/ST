149 void GUI_StyleSettings::combo_styles_changed(int idx) {
150  int new_idx = idx;
151 
152  if(_sth_changed) {
153  GlobalMessage::Answer ret = Message::question_yn(tr("There are some unsaved settings<br />Save now?"));
154 
155  if(ret == GlobalMessage::Answer::Yes) {
156  save_pressed();
157  // save was not successful
158  if(_cur_text.size() == 0) {
159  disconnect_combo_idx_changed();
160  combo_styles->setCurrentIndex(_cur_idx);
161  connect_combo_idx_changed();
162  return;
163  }
164 
165  else {
166  set_sth_changed(false);
167 
168  disconnect_combo_idx_changed();
169  combo_styles->setCurrentIndex(new_idx);
170  connect_combo_idx_changed();
171  }
172  }
173 
174  else {
175  set_sth_changed(false);
176  }
177  }
178 
179  disconnect_spinboxes();
180 
181  _cur_idx = combo_styles->currentIndex();
182  _cur_text = combo_styles->currentText();
183 
184  btn_delete->setDisabled(idx == 0);
185  btn_undo->setDisabled(idx == 0);
186 
187  RawColorStyle style = _styles[idx];
188  sb_fading_steps_lv->setValue(style.n_fading_steps_level);
189  sb_fading_steps_sp->setValue(style.n_fading_steps_spectrum);
190  sb_h_spacing_lv->setValue(style.hor_spacing_level);
191  sb_h_spacing_sp->setValue(style.hor_spacing_spectrum);
192  sb_v_spacing_lv->setValue(style.ver_spacing_level);
193  sb_v_spacing_sp->setValue(style.ver_spacing_spectrum);
194  sb_rect_height_sp->setValue(style.rect_height_spectrum);
195  sb_rect_width_lv->setValue(style.rect_width_level);
196  sb_rect_height_lv->setValue(style.rect_height_level);
197 
198  cb_col3->setChecked(style.col_list.colors.size() > 2 && style.col_list.colors[2].isValid());
199  cb_col4->setChecked(style.col_list.colors.size() > 3 && style.col_list.colors[3].isValid());

219  _cur_text = combo_styles->currentText();

224 void GUI_StyleSettings::save_pressed()
225 {
226  // we came from [0]
227  if(_cur_idx == 0 && _cur_text.size() == 0) {
228  Message::warning(tr("Please specify a name"));
229  return;
230  }
231 
232  RawColorStyle style;
233  style.n_fading_steps_level = sb_fading_steps_lv->value();
234  style.n_fading_steps_spectrum = sb_fading_steps_sp->value();
235  style.hor_spacing_level = sb_h_spacing_lv->value();
236  style.hor_spacing_spectrum = sb_h_spacing_sp->value();
237  style.ver_spacing_level = sb_v_spacing_lv->value();
238  style.ver_spacing_spectrum = sb_v_spacing_sp->value();
239  style.rect_height_spectrum = sb_rect_height_sp->value();
240  style.rect_width_level = sb_rect_width_lv->value();
241  style.rect_height_level = sb_rect_height_lv->value();
242 
243  style.col_list.name = _cur_text;
244  style.col_list.colors << _colors[0] << _colors[1];
245  if(cb_col3->isChecked()) style.col_list.colors << _colors[2];
246  if(cb_col4->isChecked()) style.col_list.colors << _colors[3];
247 
248  _db->update_raw_color_style(style);
249 
250  _styles[_cur_idx] = style;
251  set_sth_changed(false);
252 
253  disconnect_combo_idx_changed();
254 
255  int cur_idx = _cur_idx;
256  RawColorStyle tmp_style = _cur_style;
257  init();
258  combo_styles->setCurrentIndex(cur_idx);
259  _cur_idx = cur_idx;
260  _cur_text = combo_styles->currentText();
261 
262  connect_combo_idx_changed();
263  combo_styles_changed(_cur_idx);
264 
265  _cur_style = tmp_style;
266 
267  emit sig_style_update();
268 }
