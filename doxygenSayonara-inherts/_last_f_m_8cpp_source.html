187 bool LastFM::check_scrobble(quint64 pos_ms)
188 {
189  if(!_m->logged_in){
190  return false;
191  }
192 
193  if(_m->scrobbled){
194  return false;
195  }
196 
197  if(_m->md.length_ms <= 0){
198  return false;
199  }
200 
201  if(_m->old_pos == 0){
202  _m->old_pos = pos_ms;
203  _m->old_pos_difference = 0;
204  return false;
205  }
206 
207  else{
208  if(_m->old_pos > pos_ms){
209  _m->old_pos = 0;
210  }
211 
212  else if(pos_ms > _m->old_pos + 2000){
213  _m->old_pos = 0;
214  }
215 
216  else{
217  quint64 scrobble_time_ms = (quint64) (_settings->get(Set::LFM_ScrobbleTimeSec) * 1000);
218 
219  _m->old_pos_difference += (pos_ms - _m->old_pos);
220  _m->old_pos = pos_ms;
221 
222  if( (_m->old_pos_difference > scrobble_time_ms) ||
223  (_m->old_pos_difference >= ((_m->md.length_ms * 3) / 4) && _m->md.length_ms >= 1000))
224  {
225  scrobble(_m->md);
226  }
227  }
228  }
229 
230  return _m->scrobbled;
231 }

233 void LastFM::scrobble(const MetaData& md)
234 {
235  _m->scrobbled = true;
236 
237  if(!_m->active) {
238  return;
239  }
240 
241  if(!_m->logged_in){
242  return;
243  }
244 
245  LFMWebAccess* lfm_wa = new LFMWebAccess();
246  connect(lfm_wa, &LFMWebAccess::sig_response, this, &LastFM::sl_scrobble_response);
247  connect(lfm_wa, &LFMWebAccess::sig_error, this, &LastFM::sl_scrobble_error);
248 
249  time_t rawtime, started;
250  rawtime = time(nullptr);
251  struct tm* ptm = localtime(&rawtime);
252  started = mktime(ptm);
253 
254  QString artist = md.artist;
255  QString title = md.title;
256 
257  UrlParams sig_data;
258  sig_data["api_key"] = LFM_API_KEY;
259  sig_data["artist"] = artist.toLocal8Bit();
260  sig_data["duration"] = QString::number(md.length_ms / 1000).toLocal8Bit();
261  sig_data["method"] = "track.scrobble";
262  sig_data["sk"] = _m->session_key.toLocal8Bit();
263  sig_data["timestamp"] = QString::number(started).toLocal8Bit();
264  sig_data["track"] = title.toLocal8Bit();
265 
266  sig_data.append_signature();
267 
268  QByteArray post_data;
269  QString url = lfm_wa->create_std_url_post("http://ws.audioscrobbler.com/2.0/", sig_data, post_data);
270 
271  lfm_wa->call_post_url(url, post_data);
272 }

276 void LastFM::sl_similar_artists_available(IDList artist_ids) {
277  if(artist_ids.isEmpty()){
278  return;
279  }
280 
281  DatabaseConnector* db = DatabaseConnector::getInstance();
282  PlaylistHandler* plh = PlaylistHandler::getInstance();
283 
284  PlaylistConstPtr active_playlist;
285 
286  int active_idx;
287 
288  active_idx = plh->get_active_idx();
289  active_playlist = plh->get_playlist_at(active_idx);
290  const MetaDataList& v_md = active_playlist->get_playlist();
291 
292  std::random_shuffle(artist_ids.begin(), artist_ids.end());
293 
294  for( auto it=artist_ids.begin(); it != artist_ids.end(); it++ )
295  {
296  MetaDataList artist_tracks;
297 
298  db->getAllTracksByArtist(*it, artist_tracks);
299 
300  std::random_shuffle(artist_tracks.begin(), artist_tracks.end());
301 
302  // try all songs of artist
303  for(int rounds=0; rounds < artist_tracks.size(); rounds++) {
304  int rnd_track = RandomGenerator::get_random_number(0, artist_tracks.size()- 1);
305 
306  MetaData md = artist_tracks.takeAt(rnd_track);
307 
308  // two times the same track is not allowed
309  bool track_exists = std::any_of(v_md.begin(), v_md.end(), [md](const MetaData& it_md){
310  return (md.id == it_md.id);
311  });
312 
313  if(!track_exists){
314  MetaDataList v_md; v_md << md;
315 
316  plh->append_tracks(v_md, active_idx);
317  return;
318  }
319  }
320  }
321 }
