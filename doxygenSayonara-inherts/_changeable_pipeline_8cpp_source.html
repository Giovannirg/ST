79 {

94  data->element_of_interest = element;

159 static GstPadProbeReturn
160 src_blocked_remove(GstPad* pad, GstPadProbeInfo* info, gpointer data)
161 {
162  GstPad *srcpad, *sinkpad;
163  ProbeData* probe_data = (ProbeData*) data;
164 
165  gst_pad_remove_probe (pad, GST_PAD_PROBE_INFO_ID (info));
166 
167  srcpad = gst_element_get_static_pad(probe_data->element_of_interest, "src");
168  gst_pad_add_probe (srcpad,
169  (GstPadProbeType)(GST_PAD_PROBE_TYPE_BLOCK | GST_PAD_PROBE_TYPE_EVENT_DOWNSTREAM),
170  eos_probe_installed_remove,
171  probe_data,
172  NULL);
173 
174  gst_object_unref (srcpad);
175 
176  sinkpad = gst_element_get_static_pad (probe_data->element_of_interest, "sink");
177  gst_pad_send_event (sinkpad, gst_event_new_eos ());
178  gst_object_unref (sinkpad);
179 
180  return GST_PAD_PROBE_OK;
181 }
