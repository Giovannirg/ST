166 QModelIndex PlaylistItemModel::getPrevRowIndexOf(const QString& substr, int row, const QModelIndex &parent)
167 {
168  Q_UNUSED(parent)
169 
170  QString converted_string = substr;
171 
172  int len = _pl->get_count();
173  if(len < row) row = len - 1;
174 
175  // ALBUM
176  if(converted_string.startsWith(ALBUM_SEARCH))
177  {
178  converted_string.remove(ALBUM_SEARCH);
179  converted_string = converted_string.trimmed();
180 
181  for(int i=0; i<len; i++)
182  {
183  if(row - i < 0) row = len - 1;
184  int row_idx = (row - i) % len;
185 
186  QString album = _pl->at_const_ref(row_idx).album;
187  album = Library::convert_search_string(album, search_mode());
188 
189  if(album.contains(converted_string))
190  {
191  return this->index(row_idx, 0);
192  }
193  }
194  }
195 
196  //ARTIST
197  else if(converted_string.startsWith(ARTIST_SEARCH))
198  {
199  converted_string.remove(ARTIST_SEARCH);
200  converted_string = converted_string.trimmed();
201 
202  for(int i=0; i<len; i++)
203  {
204  if(row - i < 0) row = len - 1;
205  int row_idx = (row - i) % len;
206 
207  QString artist = _pl->at_const_ref(row_idx).artist;
208  artist = Library::convert_search_string(artist, search_mode());
209 
210  if(artist.contains(converted_string))
211  {
212  return this->index(row_idx, 0);
213  }
214  }
215  }
216 
217  // JUMP
218  else if(converted_string.startsWith(JUMP))
219  {
220  converted_string.remove(JUMP);
221  converted_string = converted_string.trimmed();
222  bool ok;
223  int line = converted_string.toInt(&ok);
224  if(ok && len > line) {
225  return this->index(line, 0);
226  }
227  }
228 
229  // TITLE
230  else
231  {
232  for(int i=0; i<len; i++)
233  {
234  if(row - i < 0) row = len - 1;
235  int row_idx = (row - i) % len;
236  QString title = _pl->at_const_ref(row_idx).title;
237  title = Library::convert_search_string(title, search_mode());
238 
239  if(title.contains(converted_string))
240  {
241  return this->index(row_idx, 0);
242  }
243  }
244  }
245 
246  return this->index(-1, -1);
247 }
