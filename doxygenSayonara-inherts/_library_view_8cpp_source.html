291 void LibraryView::keyPressEvent(QKeyEvent* event)
292 {
293  int key = event->key();
294 
295  Qt::KeyboardModifiers modifiers = event->modifiers();
296 
297  bool shift_pressed = (modifiers & Qt::ShiftModifier);
298  bool alt_pressed = (modifiers & Qt::AltModifier);
299  bool ctrl_pressed = (modifiers & Qt::ControlModifier);
300 
301  if((key == Qt::Key_Up || key == Qt::Key_Down))
302  {
303  if(this->selectionModel()->selection().isEmpty())
304  {
305  if(_model->rowCount() > 0) {
306  selectRow(0);
307  }
308 
309  return;
310  }
311 
312  if(ctrl_pressed){
313  event->setModifiers(Qt::NoModifier);
314  }
315  }
316 
317  SearchableTableView::keyPressEvent(event);
318  if(is_minisearcher_active()) {
319  return;
320  }
321 
322  SP::Set<int> selections = get_selected_items();
323 
324  switch(key)
325  {
326  case Qt::Key_Return:
327  case Qt::Key_Enter:
328  if(selections.isEmpty() || ctrl_pressed){
329  break;
330  }
331 
332  // standard enter
333  if(!shift_pressed && !alt_pressed)
334  {
335  if(!selections.isEmpty()){
336  int first_idx = selections.first();
337  emit doubleClicked( _model->index(first_idx, 0));
338  }
339  }
340 
341  // enter with shift
342  else if(shift_pressed && !alt_pressed) {
343  emit sig_append_clicked();
344  }
345 
346  else if(alt_pressed) {
347  emit sig_play_next_clicked();
348  }
349 
350  break;
351 
352  case Qt::Key_End:
353  this->selectRow(_model->rowCount() - 1);
354  break;
355 
356  case Qt::Key_Home:
357  this->selectRow(0);
358  break;
359 
360  default:
361  break;
362  }
363 }

365 void LibraryView::contextMenuEvent(QContextMenuEvent* event)
366 {
367  if(!_m->rc_menu) {
368  rc_menu_init();
369  }
370 
371  SP::Set<int> selections = get_selected_items();
372 
373  QPoint pos = event->globalPos();
374 
375  if(_m->type == MD::Interpretation::Tracks && selections.size() == 1)
376  {
377  _m->rc_menu->show_action(LibraryContextMenu::EntryLyrics, true);
378  }
379  else{
380  _m->rc_menu->show_action(LibraryContextMenu::EntryLyrics, false);
381  }
382 
383  bool is_right_type =
384  (_m->type == MD::Interpretation::Artists ||
385  _m->type == MD::Interpretation::Albums);
386 
387  if(is_right_type) {
388  size_t n_selections = selections.size();
389 
390  if(n_selections > 1){
391  _m->merge_menu->clear();
392 
393  for(int i : selections)
394  {
395  int id = _model->get_id_by_row(i);
396  if(id < 0){
397  n_selections--;
398  if(n_selections <= 1) {
399  break;
400  }
401 
402  continue;
403  }
404 
405  QString name = _model->get_string(i);
406  name.replace("&", "&&");
407 
408  QAction* action = new QAction(name, _m->merge_menu);
409  action->setData(id);
410  _m->merge_menu->addAction(action);
411  connect(action, &QAction::triggered, this, &LibraryView::merge_action_triggered);
412  }
413 
414  _m->merge_action->setVisible(n_selections > 1);
415  }
416  }
417 
418  rc_menu_show(pos);
419 
420  QTableView::contextMenuEvent(event);
421 }

435 void LibraryView::dropEvent(QDropEvent *event)
436 {
437  event->accept();
438  const QMimeData* mime_data = event->mimeData();
439 
440  if(!mime_data) return;
441 
442  QString text = "";
443 
444  if(mime_data->hasText()){
445  text = mime_data->text();
446  }
447 
448  // extern drops
449  if( !mime_data->hasUrls() || text.compare("tracks", Qt::CaseInsensitive) == 0) {
450  return;
451  }
452 
453  QStringList filelist;
454  for(const QUrl& url : mime_data->urls()) {
455  QString path = url.path();
456 
457  if(QFile::exists(path)) {
458  filelist << path;
459  }
460  }
461 
462  emit sig_import_files(filelist);
