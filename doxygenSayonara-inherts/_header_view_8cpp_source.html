116 
117  int n_cols = _column_headers.get_shown_columns();

128 
129  if(h->get_size_type() == ColumnHeader::SizeType::Abs) {
130  preferred_size = h->get_preferred_size_abs();
131  }
132 
133  else{

135  desired_width += h->get_preferred_size_abs();
136  }
137

144

147  target_width = desired_width;
148  view->setHorizontalScrollBarPolicy(Qt::ScrollBarAlwaysOn);
149  }
150

154
