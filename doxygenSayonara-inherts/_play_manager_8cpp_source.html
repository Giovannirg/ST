32 template<typename T, int N_ITEMS>
33 class RingBuffer {
34  private:
35  int _cur_idx;
36  int _n_items;
37  T _data[N_ITEMS];
38 
39  public:
40  RingBuffer()
41  {
42  clear();
43  }
44 
45  void clear()
46  {
47  _cur_idx = 0;
48  _n_items = 0;
49  }
50 
51  void insert(const T& item){
52  _data[_cur_idx] = item;
53  _cur_idx = (_cur_idx + 1) % N_ITEMS;
54  _n_items = std::min(N_ITEMS, _n_items + 1);
55  }
56 
57  bool has_item(const T& item) const
58  {
59  for(int i=0; i<_n_items; i++){
60  if(_data[i] == item){
61  return true;
62  }
63  }
64 
65  return false;
66  }
67 
68  int count() const
69  {
70  return _n_items;
71  }
72 
73  bool is_empty() const
74  {
75  return (_n_items == 0);
76  }
77 };

259 void PlayManager::change_track(const MetaData& md, int playlist_idx)
260 {
261  _m->md = md;
262  _m->position_ms = 0;
263  _m->cur_idx = playlist_idx;
264  _m->ring_buffer.clear();
265 
266  // initial position is outdated now and never needed again
267  if(_m->initial_position_ms > 0){
268  int old_idx = _settings->get(Set::PL_LastTrack);
269  if(old_idx != _m->cur_idx){
270  _m->initial_position_ms = 0;
271  }
272  }
273 
274  // play or stop
275  if(_m->cur_idx >= 0){
276  emit sig_track_changed(_m->md);
277  emit sig_track_idx_changed(_m->cur_idx);
278 
279  play();
280  if( (md.radio_mode() != RadioMode::Off) &&
281  _settings->get(Set::Engine_SR_Active) &&
282  _settings->get(Set::Engine_SR_AutoRecord) )
283  {
284  record(true);
285  }
286  }
287 
288  else {
289  sp_log(Log::Info) << "Playlist finished";
290  emit sig_playlist_finished();
291  stop();
292  }
293 
294  // save last track
295  if(md.db_id == 0){
296  _settings->set(Set::PL_LastTrack, _m->cur_idx);
297  }
298 
299  else{
300  _settings->set(Set::PL_LastTrack, -1);
301  }
302 
303  // show notification
304  if(_settings->get(Set::Notification_Show)){
305  if(_m->cur_idx > -1 && !_m->md.filepath().isEmpty()){
306  NotificationHandler::getInstance()->notify(_m->md);
307  }
308  }
309 }
