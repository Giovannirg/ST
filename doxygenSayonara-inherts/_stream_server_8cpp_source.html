165 // either show a popup dialog or accept directly
166 void StreamServer::new_client_request()
167 {
168  QTcpSocket* pending_socket = _m->server->nextPendingConnection();
169  if(!pending_socket) {
170  return;
171  }
172 
173  QString pending_ip = pending_socket->peerAddress().toString();
174 
175  if(_m->discmissed_ips.contains(pending_ip)){
176  reject_client(pending_socket, pending_ip);
177  _m->discmissed_ips.removeOne(pending_ip);
178  return;
179  }
180 
181  _m->pending << QPair<QTcpSocket*, QString>(pending_socket, pending_ip);
182 
183  if(_m->asking){
184  return;
185  }
186 
187  _m->asking = true;
188 
189  do{
190  pending_socket = _m->pending[0].first;
191  pending_ip = _m->pending[0].second;
192 
193  if( _settings->get(Set::Broadcast_Prompt) ){
194  if(!_m->allowed_ips.contains(pending_ip)){
195  QString question = tr("%1 wants to listen to your music.\nOK?").arg(pending_ip);
196  GlobalMessage::Answer answer = GlobalMessage::getInstance()->question(question);
197  if(answer==GlobalMessage::Answer::Yes){
198  accept_client(pending_socket, pending_ip);
199  }
200  else{
201  reject_client(pending_socket, pending_ip);
202  }
203  }
204 
205  else{
206  accept_client(pending_socket, pending_ip);
207  }
208  }
209 
210  else{
211  accept_client(pending_socket, pending_ip);
212  }
213 
214  _m->pending.pop_front();
215 
216  } while(_m->pending.size() > 0);
217 
218  _m->asking = false;
219 }
