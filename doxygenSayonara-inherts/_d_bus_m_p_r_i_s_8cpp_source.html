239 QVariantMap DBusMPRIS::MediaPlayer2::Metadata()
240 {
241  QString cover_path;
242  CoverLocation cl;
243 
244  QVariantMap map;
245  QVariant v_object_path, v_length;
246 
247 
248  int id = _md.id;
249  if(id == -1){
250  id = RandomGenerator::get_random_number(5000, 10000);
251  }
252  QDBusObjectPath object_path(QString("/org/sayonara/track") + QString::number(id));
253 
254  v_object_path.setValue<QDBusObjectPath>(object_path);
255  v_length.setValue<qlonglong>(_md.length_ms * 1000);
256 
257  cl = CoverLocation::get_cover_location(_md);
258  cover_path = cl.preferred_path();
259 
260  map["mpris:trackid"] = v_object_path;
261  map["mpris:length"] = v_length;
262  map["xesam:title"] = _md.title;
263  map["xesam:album"] = _md.album;
264  map["xesam:artist"] = QStringList() << _md.artist;
265  map["mpris:artUrl"] = QUrl::fromLocalFile(cover_path).toString();
266 
267  return map;
268 }
