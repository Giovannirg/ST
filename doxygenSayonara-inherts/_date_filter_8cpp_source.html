27 static QDateTime substract_span(const QDateTime& t, Library::DateFilter::TimeSpan span, int val)
28 {
29  switch(span)
30  {
31  case Library::DateFilter::TimeSpan::Days:
32  return t.addDays(-val);
33  case Library::DateFilter::TimeSpan::Weeks:
34  return t.addDays(-val * 7);
35  case Library::DateFilter::TimeSpan::Months:
36  return t.addMonths(-val);
37  case Library::DateFilter::TimeSpan::Years:
38  return t.addYears(-val);;
39  default:
40  return QDateTime();
41  }
42 }

208 QString Library::DateFilter::get_sql_filter(const QString& track_prefix) const
209 {
210  if(!_m->valid){
211  return QString();
212  }
213 
214  QString ret;
215  QStringList change_modes;
216  QStringList change_modes_results;
217 
218  if(_m->change_mode == ChangeMode::Modified ||
219  _m->change_mode == ChangeMode::ModifiedOrCreated)
220  {
221  if(track_prefix.isEmpty()){
222  change_modes << "modifydate";
223  }
224  else{
225  change_modes << QString(track_prefix + ".modifydate");
226  }
227  }
228 
229  if(_m->change_mode == ChangeMode::Created ||
230  _m->change_mode == ChangeMode::ModifiedOrCreated)
231  {
232  if(track_prefix.isEmpty()){
233  change_modes << "createdate";
234  }
235  else{
236  change_modes << QString(track_prefix + ".createdate");
237  }
238  }
239 
240  for(const QString& change_mode : change_modes)
241  {
242  QString change_modes_result;
243  if(_m->span_from > 0){
244  change_modes_result = change_mode + " > " + QString::number(_m->span_from);
245  }
246 
247  if(_m->span_from > 0 && _m->span_to > 0){
248  change_modes_result += " AND ";
249  }
250 
251  if(_m->span_to > 0){
252  change_modes_result += change_mode + " < " + QString::number(_m->span_to);
253  }
254 
255  change_modes_results << change_modes_result;
256  }
257 
258  if(change_modes_results.size() == 1){
259  ret = change_modes_results[0];
260  }
261 
262  else if(change_modes_results.size() == 2){
263  ret = "(" + change_modes_results[0] + ")";
264  ret += " OR ";
265  ret += "(" + change_modes_results[1] + ")";
266  }
267 
268  return ret + " ";
269 }

285 Library::DateFilter Library::DateFilter::fromString(const QString& str)
286 {
287  QStringList lst = str.split(";");
288  if(lst.size() < 4){
289  return Library::DateFilter();
290  }
291 
292  Library::DateFilter filter(lst[0]);
293  Library::DateFilter::Type type = (Library::DateFilter::Type) (lst[1].toInt());
294  if(type == Library::DateFilter::Type::Between && lst.size() < 6)
295  {
296  return Library::DateFilter();
297  }
298 
299  switch(type)
300  {
301  case Library::DateFilter::Type::NewerThan:
302  {
303  Library::DateFilter::TimeSpan span = (Library::DateFilter::TimeSpan)(lst[2].toInt());
304  quint8 value = (quint8) (lst[3].toInt());
305 
306  filter.set_newer_than(span, value);
307  }
308 
309  break;
310  case Library::DateFilter::Type::OlderThan:
311  {
312  Library::DateFilter::TimeSpan span = (Library::DateFilter::TimeSpan)(lst[2].toInt());
313  quint8 value = (quint8) lst[3].toInt();
314 
315  filter.set_older_than(span, value);
316  }
317 
318  break;
319  case Library::DateFilter::Type::Between:
320  {
321  Library::DateFilter::TimeSpan span_from = (Library::DateFilter::TimeSpan)(lst[2].toInt());
322  quint8 value_from = (quint8)(lst[3].toInt());
323  Library::DateFilter::TimeSpan span_to = (Library::DateFilter::TimeSpan)(lst[4].toInt());
324  quint8 value_to = (quint8)(lst[5].toInt());
325 
326  filter.set_between(span_from, value_from, span_to, value_to);
327  }
328 
329  break;
330  default:
331  return Library::DateFilter();
332  }
333 
334  return filter;
335 }
