61 void LibraryPluginHandler::init(const QList<LibraryContainerInterface*>& containers)
62 {
63  QString lib_dir = Helper::get_lib_path();
64  QDir plugin_dir = QDir(lib_dir);
65  QStringList dll_filenames = plugin_dir.entryList(QDir::Files);
66  QString cur_plugin = _settings->get(Set::Lib_CurPlugin);
67 
68 
69  for(LibraryContainerInterface* container : containers){
70  if(!container){
71  continue;
72  }
73 
74  sp_log(Log::Debug, this) << "Add plugin " << container->get_display_name();
75 
76  _m->libraries << container;
77  }
78 
79  for(const QString& filename : dll_filenames) {
80  QObject* raw_plugin;
81  LibraryContainerInterface* container;
82 
83  QPluginLoader loader(plugin_dir.absoluteFilePath(filename));
84 
85  raw_plugin = loader.instance();
86  if(!raw_plugin) {
87  sp_log(Log::Warning) << "Cannot load plugin: " << filename << ": " << loader.errorString();
88  loader.unload();
89  continue;
90  }
91 
92  container = dynamic_cast<LibraryContainerInterface*>(raw_plugin);
93  if(!container) {
94  loader.unload();
95  continue;
96  }
97 
98  sp_log(Log::Info) << "Found library plugin " << container->get_display_name();
99  _m->libraries << container;
100  }
101 
102  sp_log(Log::Info) << "Found " << _m->libraries.size() << " library types";
103 
104  int i=0;
105  bool found = false;
106  for(LibraryContainerInterface* container : _m->libraries )
107  {
108  if(cur_plugin == container->get_name()){
109  _m->cur_idx = i;
110  init_library(i);
111  emit sig_idx_changed(i);
112  found = true;
113  break;
114  }
115 
116  i++;
117  }
118 
119  if(!found){
120  _m->cur_idx = 0;
121  init_library(0);
122  emit sig_idx_changed(0);
123  }
124 }
