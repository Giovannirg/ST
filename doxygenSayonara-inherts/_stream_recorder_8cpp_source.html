49 static QString get_time_str()
50 {
51  QString time_str;
52  QDateTime cur = QDateTime::currentDateTime();
53 
54  QString weekday = cur.date().longDayName(cur.date().dayOfWeek() );
55  QString year = QString::number(cur.date().year());
56  QString month = cur.date().shortMonthName(cur.date().month());
57  QString day = QString("%1").arg(cur.date().day(), 2, 10, QLatin1Char('0'));
58  QString hr = QString("%1").arg(cur.time().hour(), 2, 10, QLatin1Char('0'));
59  QString min = QString("%1").arg(cur.time().minute(), 2, 10, QLatin1Char('0'));
60 
61  time_str = weekday + "_" + year + "-" + month + "-" + day + "_" + hr + "h" + min;

110 QString StreamRecorder::change_track(const MetaData& md)
111 {
112  QString sr_path;
113  QString session_path;
114  QString title;
115 
116  if(!_m->recording){
117  return "";
118  }
119 
120  if(md.title == _m->md.title) {
121  return _m->sr_recording_dst;
122  }
123 
124  bool saved = save();
125  if(saved){
126  _m->cur_idx++;
127  }
128 
129  if(!Helper::File::is_www(md.filepath())) {
130  _m->recording = false;
131  _m->sr_recording_dst = "";
132  return "";
133  }
134 
135  _m->md = md;
136  _m->md.year = QDateTime::currentDateTime().date().year();
137  _m->md.track_num = _m->cur_idx;
138 
139  title = QString("%1 - %2 - %3")
140  .arg(_m->cur_idx, 3, 10, QLatin1Char('0'))
141  .arg(md.artist)
142  .arg(md.title);
143 
144  title.replace("/", "_");
145  title.replace("\\", "_");
146  title.replace(":", "");
147 
148  sr_path = _settings->get(Set::Engine_SR_Path);
149  session_path = check_session_path(sr_path);
150 
151  if(session_path.isEmpty()){
152  _m->sr_recording_dst = "";
153  _m->session_playlist_name = "";
154  _m->recording = false;
155  }
156 
157  else{
158  _m->session_playlist_name = session_path + "/playlist.m3u";
159  _m->sr_recording_dst = session_path + "/" + title + ".mp3";
160  }
161 
162  return _m->sr_recording_dst;
163 }
