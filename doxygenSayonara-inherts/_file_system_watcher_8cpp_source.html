41 void FileSystemWatcher::run()
42 {
43  _may_run = true;
44  _refresh = false;
45  _waiting = false;
46 
47  _indexed_files = index_files(_library_path);
48 
49  while(_may_run)
50  {
51  bool changed = false;
52 
53  if(_refresh)
54  {
55  if(!_may_run){
56  break;
57  }
58 
59  _indexed_files = index_files(_library_path);
60  _refresh = false;
61  _waiting = false;
62  }
63 
64  if(!_waiting)
65  {
66  if(!_may_run){
67  break;
68  }
69 
70  QStringList actual_files = index_files(_library_path);
71 
72  if(actual_files.size() != _indexed_files.size()){
73  changed = true;
74  }
75 
76  else
77  {
78  auto it1 = _indexed_files.begin();
79  auto it2 = actual_files.begin();
80 
81  for(; it1 != _indexed_files.end(); it1++, it2++){
82  if(!_may_run){
83  break;
84  }
85 
86  if(it1->compare(*it2) != 0){
87  changed = true;
88  break;
89  }
90  }
91  }
92  }
93 
94  if(changed && _may_run){
95  _waiting = true;
96  emit sig_changed();
97  }
98 
99  for(int i=0; i<SleepingTimeSec; i++){
100  if(!_may_run){
101  break;
102  }
103 
104  QThread::msleep(1000);
105  }
106 
107  if(!_may_run){
108  break;
109  }
110  }
111 
112  sp_log(Log::Info) << "Watcher finished";
113 }
