197 void Bookmarks::pos_changed_ms(quint64 pos_ms)
198 {
199  _m->cur_time = (quint32) (pos_ms / 1000);
200 
201  if( _m->cur_time >= _m->loop_end &&
202  _m->loop_end != 0)
203  {
204  jump_prev();
205  return;
206  }
207 
208  if(_m->bookmarks.isEmpty()){
209  return;
210  }
211 
212  _m->prev_idx=-1;
213  _m->next_idx=-1;
214 
215 
216  int i=0;
217  for(Bookmark& bookmark : _m->bookmarks){
218  quint32 time = bookmark.get_time();
219 
220  if(time + 2 < _m->cur_time){
221  _m->prev_idx = i;
222  }
223

225  if(_m->next_idx == -1){
226  _m->next_idx = i;
227  break;

232  }
