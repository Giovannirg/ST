134 void LibraryImporter::copy_thread_finished()
135 {
136  CopyThread* copy_thread = static_cast<CopyThread*>(sender());
137 
138  MetaDataList v_md = copy_thread->get_copied_metadata();
139 
140  emit_status(ImportStatus::Sleeping);
141 
142  // no tracks were copied or rollback was finished
143  if(v_md.isEmpty()) {
144  emit_status(ImportStatus::NoTracks);
145  copy_thread->deleteLater();
146  return;
147  }
148 
149  // copy was cancelled
150  sp_log(Log::Debug, this) << "Copy folder thread finished " << _m->copy_thread->was_cancelled();
151  if(copy_thread->was_cancelled()) {
152  copy_thread->set_mode(CopyThread::Mode::Rollback);
153  copy_thread->start();
154 
155  emit_status(ImportStatus::Rollback);
156  return;
157  }
158 
159  // store to db
160  bool success = _m->db->storeMetadata(v_md);
161  int n_files_copied = copy_thread->get_n_copied_files();
162  int n_files_to_copy = _m->import_cache.get_files().size();
163 
164  // error and success messages
165  if(success) {
166  _m->db->clean_up();
167 
168  QString str = "";
169  if(n_files_to_copy == n_files_copied){
170  str = tr("All files could be imported");
171  }
172 
173  else{
174  str = tr("%1 of %2 files could be imported")
175  .arg(n_files_copied)
176  .arg(n_files_to_copy);
177  }
178 
179  Message::info(str);
180 
181  emit_status(ImportStatus::Imported);
182 
183  MetaDataChangeNotifier::getInstance()->change_metadata(MetaDataList(), MetaDataList());
184  }
185 
186  else{
187  QString warning = tr("Sorry, but tracks could not be imported") +
188  "<br />" +
189  tr("Please use the import function of the file menu<br /> or move tracks to library and use 'Reload library'");
190 
191  Message::warning(warning);
192  }
193 
194  copy_thread->deleteLater();
195 }
