374 void LocalLibrary::merge_artists(const SP::Set<ArtistID>& artist_ids, ArtistID target_artist)
375 {
376  if(artist_ids.isEmpty()) {
377  return;
378  }
379 
380  if(target_artist < 0){
381  sp_log(Log::Warning, this) << "Cannot merge artist: Target artist id < 0";
382  return;
383  }
384 
385  bool show_album_artists = _settings->get(Set::Lib_ShowAlbumArtists);
386 
387  Artist artist;
388  bool success = _m->db->getArtistByID(target_artist, artist);
389  if(!success){
390  return;
391  }
392 
393  MetaDataList v_md;
394 
395  get_all_tracks_by_artist(artist_ids.toList(), v_md, _filter, _sortorder);
396  _m->tag_edit->set_metadata(v_md);
397 
398  for(int idx=0; idx<v_md.size(); idx++)
399  {
400  MetaData md = v_md[idx];
401  if(show_album_artists){
402  md.set_album_artist(artist.name, artist.id);
403  }
404 
405  else {
406  md.artist_id = artist.id;
407  md.artist = artist.name;
408  }
409 
410  _m->tag_edit->update_track(idx, md);
411  }
412 
413  _m->tag_edit->commit();
414 }
