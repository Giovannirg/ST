40 create_colorstyle(ColorStyle &style, const ColorList &clist_active, int n_rects, int n_fading_steps) {
41  style.style.clear();
42 
43  QHash<int, QColor> map_col_active;
44 
45  // compute color of each rect
46  for(int i=0; i<n_rects; i++) {
47  insertColorOfRect(i, n_rects, clist_active, map_col_active);
48  }
49 
50  // compute active to inactive color
51  QList<float> borders;
52  borders << 0.0 << 1.0;
53 
54  // run through rect
55  for(int idx_rect=0; idx_rect < n_rects; idx_rect++) {
56  QColor col_active = map_col_active.value(idx_rect);
57 
58  // fadeout
59  QHash<int, QColor> fading_map;
60 
61  ColorList col_list;
62  col_list.colors << QColor(0,0,0,50) << col_active.darker();
63 
64  // run through step
65  for(int idx_step=0; idx_step<=n_fading_steps; idx_step++){
66  insertColorOfRect(idx_step, n_fading_steps + 1, col_list, fading_map);
67  }
68 
69  fading_map[-1] = col_active;
70 
71  style.style << fading_map;
72  }
73

78 void EngineColorStyleChooser::insertColorOfRect(int bin, int n_bins, const ColorList& colorlist, QHash<int, QColor>& map) {
79  QColor col;
80  QList<float> borders;
81 
82  if(colorlist.colors.size() == 4) {
83  borders = borders_4;
84  }
85 
86  else if(colorlist.colors.size() == 3){
87  borders = borders_3;
88  }
89 
90  else if(colorlist.colors.size() == 2){
91  borders = borders_2;
92  }
93 
94  float x = (bin * 1.0f) / n_bins;
95  int i=0;
96  int r, g, b, a;
97 
98  if(bin == 0) {
99  map[bin] = colorlist.colors[0];
100  return;
101  }
102 
103  while(x > borders[i]) {
104  i++;
105  }
106 
107  float dx = (borders[i] - borders[i-1]);
108 
109  float dy = colorlist.colors[i].red() - colorlist.colors[i-1].red();
110  r = (int) (( dy * (x-borders[i-1]) ) / dx + colorlist.colors[i-1].red());
111 
112  dy = colorlist.colors[i].green() - colorlist.colors[i-1].green();
113  g = (int) (( dy * (x-borders[i-1])) / dx + colorlist.colors[i-1].green());
114 
115  dy = colorlist.colors[i].blue() - colorlist.colors[i-1].blue();
116  b = (int) ((dy * (x-borders[i-1])) / dx + colorlist.colors[i-1].blue());
117 
118  dy = colorlist.colors[i].alpha() - colorlist.colors[i-1].alpha();
119  a = (int) ((dy * (x-borders[i-1])) / dx + colorlist.colors[i-1].alpha());
120 
121  col.setRed(r);
122  col.setGreen(g);
123  col.setBlue(b);
124  col.setAlpha(a);
125 
126  map[bin] = col;
127 }
