64 void GUI_LanguageChooser::renew_combo()
65 {
66  if(!is_ui_initialized()){
67  return;
68  }
69 
70  QString lang_setting = _settings->get(Set::Player_Language);
71  sp_log(Log::Info) << "Language setting = " << lang_setting;
72  QDir dir(Helper::get_share_path("translations"));
73 
74  QStringList filters;
75  filters << "*.qm";
76  QStringList files = dir.entryList(filters);
77 
78  ui->combo_lang->clear();
79 
80  int i=0;
81  for(const QString& file : files) {
82  QRegExp re(".*lang_(.*)\\.qm");
83  re.setMinimal(true);
84 
85  QString country_code;
86  if(re.indexIn(file) >= 0){
87  country_code = re.cap(1).toLower();
88  }
89 
90  else{
91  continue;
92  }
93 
94  if(country_code.compare("mx") == 0){
95  continue;
96  }
97 
98  QString icon_path = Helper::get_share_path(
99  "translations/icons/" + country_code + ".png"
100  );
101 
102  QString language_name = _map.value(country_code);
103 
104  if(language_name.size() > 0){
105  ui->combo_lang->addItem(QIcon(icon_path), language_name, file);
106  }
107 
108  else{
109  ui->combo_lang->addItem(file, file);
110  }
111 
112  if(file.contains(lang_setting, Qt::CaseInsensitive)){
113  ui->combo_lang->setCurrentIndex(i);
114  }
115 
116  i++;
117  }
118 }
