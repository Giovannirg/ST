149 void SoundcloudLibrary::insert_tracks(const MetaDataList& v_md, const ArtistList& artists, const AlbumList& albums){
150  Artist artist_tmp;
151  Album album_tmp;
152 
153  for(const Artist& artist : artists){
154  if(!_scd->getArtistByID(artist.id, artist_tmp) || artist.id != artist_tmp.id){
155  int id = _scd->insertArtistIntoDatabase(artist);
156  sp_log(Log::Debug, this) << "Insert artist into database: " << id;
157  }
158  }
159 
160  for(const Album& album : albums){
161  sp_log(Log::Debug, this) << "Try to fetch album " << album.id;
162  if(!_scd->getAlbumByID(album.id, album_tmp) || album.id != album_tmp.id){
163  int id = _scd->insertAlbumIntoDatabase(album);
164  sp_log(Log::Debug, this) << "Insert album into database: " << id;
165  }
166 
167  else{
168  album_tmp.print();
169  }
170  }
171 
172  if(!_scd->getAlbumByID(-1, album_tmp)){
173  Album album;
174  album.name = "None";
175  album.id = 0;
176 
177  _scd->insertAlbumIntoDatabase(album);
178  }
179 
180  _scd->storeMetadata(v_md);
181 
182  AbstractLibrary::insert_tracks(v_md);
183 }

186 void SoundcloudLibrary::artists_fetched(const ArtistList& artists){
187  for(const Artist& artist : artists){
188  SoundcloudDataFetcher* fetcher;
189 
190  if(artist.id <= 0) {

192  }
193 
194  _scd->updateArtist(artist);

197

199  this, &SoundcloudLibrary::albums_fetched);
200 
201  connect(fetcher, &SoundcloudDataFetcher::sig_tracks_fetched,
202  this, &SoundcloudLibrary::tracks_fetched);
203 
204  fetcher->get_tracks_by_artist(artist.id);
205  }
206 
207  refetch();
208 
209  sender()->deleteLater();
