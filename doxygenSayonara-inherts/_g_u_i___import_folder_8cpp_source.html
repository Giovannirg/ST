92 void GUI_ImportFolder::set_status(LibraryImporter::ImportStatus status)
93 {
94  ui->pb_progress->hide();
95  ui->lab_status->show();
96 
97 
98  bool thread_active = false;
99 
100  switch(status){
101  case LibraryImporter::ImportStatus::Caching:
102  ui->lab_status->setText(tr("Loading tracks..."));
103  thread_active = true;
104  show();
105  break;
106 
107  case LibraryImporter::ImportStatus::Importing:
108  ui->lab_status->setText(tr("Importing..."));
109  thread_active = true;
110  break;
111 
112  case LibraryImporter::ImportStatus::Imported:
113  ui->lab_status->setText(tr("Finished"));
114  close();
115  break;
116 
117  case LibraryImporter::ImportStatus::Cancelled:
118  ui->lab_status->setText(tr("Cancelled"));
119  close();
120  break;
121 
122  case LibraryImporter::ImportStatus::NoTracks:
123  ui->lab_status->setText(tr("No tracks"));
124  break;
125 
126  case LibraryImporter::ImportStatus::Rollback:
127  ui->lab_status->setText(tr("Rollback"));
128  thread_active = true;
129  break;
130 
131  default:
132  break;
133  }
134 
135  if(thread_active){
136  ui->btn_cancel->setText(tr("Cancel"));
137  }
138 
139  else{
140  ui->btn_cancel->setText(tr("Close"));
141  }
142 }

188 void GUI_ImportFolder::choose_dir()
189 {
190  QString lib_path = _settings->get(Set::Lib_Path);
191  QString dialog_title = tr("Choose target directory");
192  QString dir =
193  QFileDialog::getExistingDirectory( this,
194  dialog_title,
195  lib_path,
196  QFileDialog::ShowDirsOnly
197  );
198 
199  if(dir.isEmpty()){
200  ui->le_directory->clear();
201  return;
202  }
203 
204  if(!dir.contains(lib_path)) {
205  Message::warning(tr("%1<br />is no library directory").arg(dir));
206  ui->le_directory->clear();
207  return;
208  }
209 
210  dir.replace(lib_path, "");
211  while(dir.startsWith(QDir::separator())){
212  dir.remove(0, 1);
213  }
214 
215  while(dir.endsWith(QDir::separator())){
216  dir.remove( dir.size() - 1, 1);
217  }
218 
219  ui->le_directory->setText(dir);
220 }
