79 bool CoverFetchThread::start()
80 {
81  _m->may_run = true;
82 
83  if(_m->cl.has_search_urls()){
84  _m->url = _m->cl.search_urls().first();
85  _m->cl.remove_first_search_url();
86  }
87 
88  else {
89  return false;

92  CoverFetchManager* cfm = CoverFetchManager::getInstance();

104  more();
105  }
106 
107  else

109  AsyncWebAccess* awa = new AsyncWebAccess(this);
110  awa->setObjectName(_m->acf->get_keyword());
111  awa->set_behavior(AsyncWebAccess::Behavior::AsSayonara);
112  connect(awa, &AsyncWebAccess::sig_finished, this, &CoverFetchThread::content_fetched);
113 
114  _m->active_connections << awa;
115  awa->run(_m->url, Timeout);
116  }
117 
118  return true;
119 }

122 bool CoverFetchThread::more()
123 {
124  if(_m->may_run == false){
125  return false;
126  }
127 
128  // we have all our covers
129  if(_m->n_covers == _m->n_covers_found){
130  emit sig_finished(true);
131  return true;
132  }
133 
134  // we have no more addresses and not all our covers
135  if(_m->addresses.isEmpty())
136  {
137  bool success = start();
138  if(!success) {
139  emit sig_finished(false);
140  }
141 
142  return success;
143  }
144 
145 
146  QString address = _m->addresses.takeFirst();
147  AsyncWebAccess* awa = new AsyncWebAccess(this);
148  awa->set_behavior(AsyncWebAccess::Behavior::AsBrowser);
149 
150  if(_m->n_covers == 1) {
151  connect(awa, &AsyncWebAccess::sig_finished, this, &CoverFetchThread::single_image_fetched);
152  }
153 
154  else {
155  connect(awa, &AsyncWebAccess::sig_finished, this, &CoverFetchThread::multi_image_fetched);
156  }
157 
158  awa->run(address, Timeout);
159  _m->active_connections << awa;
160 
161  return true;
162 }

175 void CoverFetchThread::content_fetched()
176 {
177  AsyncWebAccess* awa = static_cast<AsyncWebAccess*>(sender());
178  _m->active_connections.removeAll(awa);
179 
180  if(awa->objectName() == _m->acf->get_keyword()) {
181  if(awa->status() == AsyncWebAccess::Status::GotData)
182  {
183  QByteArray website = awa->data();
184  _m->addresses = _m->acf->calc_addresses_from_website(website);
185  }
186  }
187 
188  awa->deleteLater();

192 void CoverFetchThread::single_image_fetched()

196

203  _m->n_covers_found++;

210 
211  else {

213  if(!more()){
