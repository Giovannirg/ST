116 void IcyWebAccess::connected()
117 {
118  QString user_agent = QString("Sayonara/") + SAYONARA_VERSION;
119  QByteArray data(
120  "GET " + _m->concat_dir_and_filename().toLocal8Bit() + " HTTP/1.1\r\n"
121  "User-Agent: " + user_agent.toLocal8Bit() + "\r\n"
122  "Connection: Keep-Alive\r\n"
123  "Accept-Encoding: gzip, deflate\r\n"
124  "Accept-Language: en-US,*\r\n"
125  "Host: " +
126  _m->hostname.toLocal8Bit() + ":" +
127  QString::number(_m->port).toLocal8Bit() + "\r\n\r\n"
128  );
129 
130  sp_log(Log::Develop, this) << data;
131 
132  qint64 bytes_written = _m->tcp->write(data.data(), data.size());

136  _m->status = IcyWebAccess::Status::WriteError;
