119 QModelIndex SearchViewFunctionality::get_match_index(const QString& str, SearchDirection direction) const
120 {
121  QModelIndex idx;
122  if(str.isEmpty()) {
123  return idx;
124  }
125 
126  if(!_m->search_model) {
127  return idx;
128  }
129 
130  Library::SearchModeMask search_mode = _m->settings->get(Set::Lib_SearchMode);
131  QMap<QChar, QString> extra_triggers = _m->search_model->getExtraTriggers();
132 
133  QString converted_string = Library::convert_search_string(str, search_mode, extra_triggers.keys());
134 
135  switch(direction)
136  {
137  case SearchDirection::First:
138  idx = _m->search_model->getFirstRowIndexOf(converted_string);
139  break;
140  case SearchDirection::Next:
141  idx = _m->search_model->getNextRowIndexOf(converted_string, _m->cur_idx + 1);
142  break;
143  case SearchDirection::Prev:
144  idx = _m->search_model->getPrevRowIndexOf(converted_string, _m->cur_idx - 1);
145  break;
146  }
147 
148  return idx;
149 }

152 void SearchViewFunctionality::select_match(const QString &str, SearchDirection direction)
153 {
154  QModelIndex idx = get_match_index(str, direction);
155  if(!idx.isValid()){
156  _m->cur_idx = -1;
157  return;

159

161 
162  SP::Set<int> indexes(_m->cur_idx);
163 
164  switch(selection_type()){
165  case SayonaraSelectionView::SelectionType::Rows:
166  select_rows(indexes);
167  break;
168  case SayonaraSelectionView::SelectionType::Columns:
169  select_columns(indexes);
170  break;
171  case SayonaraSelectionView::SelectionType::Items:
172  select_items(indexes);
173  break;
174  }
175 
176  this->set_current_index(_m->cur_idx);
177 
178  _m->view->scrollTo(idx);
179 }
