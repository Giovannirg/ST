181 int StdPlaylist::calc_shuffle_track()
182 {
183  if(metadata().size() <= 1){
184  return -1;
185  }
186 
187  QList<int> left_tracks;
188  int i, left_tracks_idx;
189  RandomGenerator rnd;
190 
191  // check all tracks played
192  i=0;
193  for(MetaData& md : metadata()){
194  if(!md.played){
195  left_tracks << i;
196  }
197 
198  i++;
199  }
200 
201  // no random track to play
202  if(left_tracks.isEmpty()){
203  if(Playlist::Mode::isActiveAndEnabled(_playlist_mode.repAll())){
204  return rnd.get_number(0, metadata().size() -1);
205  }
206 
207  else{
208  return -1;
209  }
210  }
211 
212  left_tracks_idx = rnd.get_number(0, left_tracks.size() - 1);
