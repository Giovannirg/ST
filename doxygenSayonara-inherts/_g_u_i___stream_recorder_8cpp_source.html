86 void GUI_StreamRecorder::commit()
87 {
88  _settings->set(Set::Engine_SR_Active, ui->cb_activate->isChecked());
89 
90  if(!ui->le_path->isEnabled()){
91  return;
92  }
93 
94  QString str = ui->le_path->text();
95  if(!QFile::exists(str))
96  {
97  bool create_success = QDir::root().mkpath(str);
98  if(!create_success)
99  {
100  QString sr_path = _settings->get(Set::Engine_SR_Path);
101  ui->le_path->setText(sr_path);
102 
103  Message::warning(str + tr(" could not be created\nPlease choose another folder"), tr("Stream recorder"));
104  }
105  }
106 
107  _settings->set(Set::Engine_SR_Path, str);
108  _settings->set(Set::Engine_SR_AutoRecord, ui->cb_auto_rec->isChecked());
109  _settings->set(Set::Engine_SR_SessionPath, ui->cb_create_session_path->isChecked());
110 }
