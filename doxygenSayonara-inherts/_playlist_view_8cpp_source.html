231 void PlaylistView::handle_drop(QDropEvent* event)
232 {
233  const QMimeData* mimedata = event->mimeData();
234  PlaylistHandler* plh = PlaylistHandler::getInstance();
235 
236  MetaDataList v_md;
237  QStringList www_playlists;
238  QString src;
239 
240  int row = _delegate->get_drag_index();
241  clear_drag_drop_lines(row);
242 
243  if(!mimedata){
244  return;
245  }
246 
247  src = mimedata->objectName();
248  bool inner_drag_drop = (src.compare("inner") == 0);
249  if(inner_drag_drop) {
250  bool copy = (event->keyboardModifiers() & Qt::ControlModifier);
251  handle_inner_drag_drop(row, copy);
252  return;
253  }
254 
255  const CustomMimeData* custom_mimedata = dynamic_cast<const CustomMimeData*>(mimedata);
256 
257  if(custom_mimedata){
258  if( custom_mimedata->hasText() &&
259  custom_mimedata->hasMetaData())
260  {
261  v_md = custom_mimedata->getMetaData();
262  }
263  }
264 
265  else if( mimedata->hasUrls()) {
266  DirectoryReader reader;
267  reader.set_filter(Helper::get_soundfile_extensions());
268 
269  for(const QUrl& url : mimedata->urls()) {
270  if(url.isLocalFile()){
271  QStringList file_list;
272  MetaDataList v_md_tmp;
273  file_list << url.toLocalFile();
274  v_md_tmp = reader.get_md_from_filelist(file_list);
275  v_md << v_md_tmp;
276  }
277 
278  else if(Helper::File::is_playlistfile(url.toString())){
279  www_playlists << url.toString();
280  }
281  }
282  }
283 
284  if(!v_md.isEmpty()) {
285  plh->insert_tracks(v_md, row+1, plh->get_current_idx());
286  }
287 
288  if(!www_playlists.isEmpty()){
289  this->setEnabled(false);
290  if(!_progress){
291  _progress = new SayonaraLoadingBar(this);
292  }
293 
294  _progress->show();
295 
296  _async_drop_index = row;
297  StreamParser* stream_parser = new StreamParser();
298  if(mimedata->hasText()){
299  stream_parser->set_cover_url(mimedata->text());
300  }
301 
302  connect(stream_parser, &StreamParser::sig_finished, this, &PlaylistView::handle_async_drop);
303  stream_parser->parse_streams(www_playlists);
304  }
305 }

325 void PlaylistView::handle_inner_drag_drop(int row, bool copy)
326 {
327  SP::Set<int> cur_selected_rows, new_selected_rows;
328  int n_lines_before_tgt = 0;
329 
330  cur_selected_rows = get_selected_items();
331 
332  if( cur_selected_rows.contains(row) ) {
333  return;
334  }
335 
336  if(copy){
337  _model->copy_rows(cur_selected_rows, row + 1);
338  emit sig_time_changed();
339  }
340 
341  else {
342  _model->move_rows(cur_selected_rows, row + 1);
343  n_lines_before_tgt = std::count_if(cur_selected_rows.begin(), cur_selected_rows.end(), [&row](int sel){
344  return sel < row;
345  });
346  }
347 
348  for(int i=row; i<row + (int) cur_selected_rows.size(); i++){
349  new_selected_rows.insert(i - n_lines_before_tgt + 1);
350  }
351 
352  this->select_rows( new_selected_rows );
353 }
