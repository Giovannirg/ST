74 bool compare_md(const MetaData& md1, const MetaData& md2)
75 {
76  if(md1.genres.count() != md2.genres.count()){
77  return false;
78  }
79 
80  auto it1 = md1.genres.begin();
81  auto it2 = md2.genres.begin();
82  for(int i=0; i<md1.genres.count(); i++, it1++, it2++){
83  if(!it1->is_equal(*it2)){
84  return false;
85  }
86  }
87 
88  return (md1.title == md2.title &&
89  md1.album == md2.album &&
90  md1.artist == md2.artist &&
91  md1.year == md2.year &&
92  md1.rating == md2.rating &&
93  md1.discnumber == md2.discnumber &&
94  md1.track_num == md2.track_num &&
95  md1.album_artist() == md2.album_artist() &&
96  md1.album_artist_id() == md2.album_artist_id()
97  );
98 }

100 int ReloadThread::get_and_save_all_files(const QHash<QString, MetaData>& md_map_lib)
101 {
102  QString library_path = _m->library_path;
103 
104  if(library_path.isEmpty() || !QFile::exists(library_path)) {
105  return 0;
106  }
107 
108  DatabaseConnector* db = _m->db;
109  QDir dir(library_path);
110 
111  MetaDataList v_md_to_store;
112  QStringList files = get_files_recursive (dir);
113 
114  int n_files = files.size();
115  int cur_idx_files=0;
116 
117  for(const QString& filepath : files){
118  bool file_was_read = false;
119  MetaData md(filepath);
120  const MetaData& md_lib = md_map_lib[filepath];
121 
122  int percent = (cur_idx_files++ * 100) / n_files;
123  emit sig_reloading_library(Lang::get(Lang::ReloadLibrary).triplePt(), percent);
124 
125  if(md_lib.id >= 0){
126  if(_m->quality == Library::ReloadQuality::Fast){
127  continue;
128  }
129 
130  file_was_read = Tagging::getMetaDataOfFile(md, Tagging::Quality::Dirty);
131 
132  if(!file_was_read){
133  continue;
134  }
135 
136  if( md_lib.length_ms > 1000 && md_lib.length_ms < 3600000 && compare_md(md, md_lib)){
137  continue;
138  }
139  }
140 
141  file_was_read = Tagging::getMetaDataOfFile(md, Tagging::Quality::Quality);
142 
143  if(file_was_read){
144  v_md_to_store << md;
145 
146  if(v_md_to_store.size() >= N_FILES_TO_STORE){
147  db->storeMetadata(v_md_to_store);
148  v_md_to_store.clear();
149  }
150  }
151  }
152 
153  if(!v_md_to_store.isEmpty()){
154  db->storeMetadata(v_md_to_store);
155  v_md_to_store.clear();
156  }
157 
158  db->addAlbumArtists();
159  db->createIndexes();
160  db->clean_up();
161 
162  return v_md_to_store.size();
163 }

166 QStringList ReloadThread::get_files_recursive(QDir base_dir)
167 {
168  QStringList ret;
169  QString message = tr("Reading files from file system") + "... ";
170  emit sig_reloading_library(message, 0);
171 
172  QStringList soundfile_exts = Helper::get_soundfile_extensions();
173  QStringList sub_dirs;
174  QStringList sub_files;
175 
176  sub_dirs = base_dir.entryList(QDir::Dirs | QDir::NoDotAndDotDot);
177 
178  for(const QString& dir : sub_dirs) {
179  bool success = base_dir.cd(dir);
180 
181  if(!success){
182  continue;
183  }
184 
185  ret << get_files_recursive(base_dir);
186 
187  base_dir.cdUp();
188  }
189 
190  sub_files = base_dir.entryList(soundfile_exts, QDir::Files);
191  if(sub_files.isEmpty()){
192  return ret;
193  }
194 
195  ret << process_sub_files(base_dir, sub_files);
196 
197  return ret;
198 }

245 void ReloadThread::run()
246 {
247  if(_m->library_path.isEmpty())
248  {
249  sp_log(Log::Warning, this) << "No Library path given";
250  return;
251  }
252 
253  if(_m->running){
254  return;
255  }
256 
257  DatabaseConnector* db = _m->db;
258 
259  _m->running = true;
260  _m->paused = false;
261 
262  MetaDataList v_md, v_to_delete;
263  QHash<QString, MetaData> v_md_map;
264 
265  emit sig_reloading_library(tr("Delete orphaned tracks..."), 0);
266 
267  db->deleteInvalidTracks();
268  db->getAllTracks(v_md);
269 
270  sp_log(Log::Debug, this) << "Have " << v_md.size() << " tracks";
271 
272  // find orphaned tracks in library && delete them
273  for(const MetaData& md : v_md){
274  if(!Helper::File::check_file(md.filepath())) {
275  v_to_delete << std::move(md);
276  }
277 
278  else{
279  v_md_map[md.filepath()] = md;
280  }
281  }
282 
283  if(!v_to_delete.isEmpty()){
284  db->deleteTracks(v_to_delete);
285  }
286 
287  get_and_save_all_files(v_md_map);
288 
289  _m->paused = false;
290  _m->running = false;
291 }
