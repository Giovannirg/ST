133 void SoundcloudDataFetcher::tracks_fetched()
134 {
135  MetaDataList v_md;
136  ArtistList artists;
137  AsyncWebAccess* awa = static_cast<AsyncWebAccess*>(sender());
138 
139  if(awa->status() != AsyncWebAccess::Status::GotData) {
140  awa->deleteLater();
141  return;
142  }
143 
144  QByteArray data = awa->data();
145  SoundcloudJsonParser parser(data);
146  parser.parse_tracks(artists, v_md);
147 
148  for(const MetaData& md : v_md){
149  if(!_m->playlist_tracks.contains(md.id)){
150  _m->playlist_tracks << md;
151  }
152  }
153 
154  for(const Artist& artist : artists){
155  if(!_m->artists.contains(artist.id)){
156  _m->artists << artist;
157  }
158  }
159 
160  emit sig_playlists_fetched(_m->playlists);
161  emit sig_tracks_fetched(_m->playlist_tracks);
162  emit sig_ext_artists_fetched(_m->artists);
163 
164  awa->deleteLater();
165 }
