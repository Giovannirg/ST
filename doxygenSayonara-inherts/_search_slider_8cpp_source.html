40 bool SearchSlider::event(QEvent *e)
41 {
42  QWheelEvent* we;
43  int delta_val = 5;
44 
45  switch(e->type()){
46  case QEvent::Wheel:
47 
48  if(!this->isEnabled()) {
49  break;
50  }
51 
52  we = (QWheelEvent*) e;
53  if(we->modifiers() & Qt::ShiftModifier){
54  delta_val = 10;
55  }
56 
57  else if(we->modifiers() & Qt::AltModifier){
58  delta_val = 50;
59  }
60 
61 
62  if(we->delta() > 0){
63  setValue(value() + delta_val);
64  }
65  else{
66  setValue(value() - delta_val);
67  }
68 
69  emit_new_val(value());
70  break;
71 
72  default:
73  break;
74  }
75 
76  return SayonaraSlider::event(e);
77 }
