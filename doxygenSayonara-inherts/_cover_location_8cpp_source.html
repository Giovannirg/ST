258 CoverLocation Get_cover_location(int album_id, quint8 db_id)
259 {
260  if(album_id < 0) {
261  return CoverLocation::getInvalidLocation();
262  }
263 
264  Album album;
265  MetaDataList v_md;
266  LibraryDatabase* db = DB::getInstance(db_id);
267 
268  bool success = db->getAlbumByID(album_id, album, true);
269 
270  if(!success) {
271  return CoverLocation::getInvalidLocation();
272  }
273 
274  CoverLocation cl = CoverLocation::get_cover_location(album);
275 
276  db->getAllTracksByAlbum(album_id, v_md);
277  for(const MetaData& md : v_md){
278  QStringList local_paths = LocalCoverSearcher::get_local_cover_paths_from_filename(md.filepath());
279  for(const QString& local_path : local_paths){
280  cl.add_local_path(local_path);
281  }
282 
283  if(!cl.local_paths().isEmpty()){
284  break;
285  }
286  }
287 
288  return cl;
