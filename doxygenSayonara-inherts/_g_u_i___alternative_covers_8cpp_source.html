177 void GUI_AlternativeCovers::apply_clicked()
178 {
179  if(_m->cur_idx == -1) {
180  return;
181  }
182 
183  RowColumn rc = _m->model->cvt_2_row_col(_m->cur_idx);
184 
185  QModelIndex idx = _m->model->index(rc.row, rc.col);
186 
187  if(!idx.isValid()) {
188  return;
189  }
190 
191  QString cover_path = _m->model->data(idx, Qt::UserRole).toString();
192  QFile file(cover_path);
193 
194  if(!file.exists()) {
195  Message::warning(tr("This cover does not exist"));
196  return;
197  }
198 
199  QImage img(cover_path);
200  if(img.isNull()){
201  return;
202  }
203 
204  img.save(_m->cover_location.cover_path());
205 
206  emit sig_cover_changed(_m->cover_location);
207 }

284 void GUI_AlternativeCovers::open_file_dialog()
285 {

296  dir.setNameFilters(filters);
297

299  QFile::remove(dir.absoluteFilePath(f));
300  }
301 
302  QStringList lst = QFileDialog::getOpenFileNames(this,
303  tr("Open image files"),
304  _m->last_path,
305  filters.join(" "));
306  if(lst.isEmpty())
307  {
308  return;
309  }
310 
311  reset_model();
312 
313  int idx = 0;
314  for(const QString& path : lst)
315  {
316  RowColumn rc = _m->model->cvt_2_row_col( idx );
317  QModelIndex model_idx = _m->model->index(rc.row, rc.col);
318 
319  _m->model->setData(model_idx, path);
320 
321  idx ++;
322  }
323 }
