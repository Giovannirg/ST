134 bool MiniSearcher::is_initiator(QKeyEvent* event) const
135 {
136  QString text = event->text();
137 
138  if(event->modifiers() & Qt::ControlModifier){
139  return false;
140  }
141 
142  if(text.isEmpty()){
143  return false;
144  }
145 
146  if(text[0].isLetterOrNumber()){
147  return true;
148  }
149 
150  if(_m->triggers.contains(text[0]) ){
151  return true;
152  }
153 
154  return false;
155 }

252 void MiniSearcher::init(QString text)
253 {
254  QScrollBar* v_scrollbar = _m->parent->verticalScrollBar();
255  QScrollBar* h_scrollbar = _m->parent->horizontalScrollBar();
256 
257  int sb_width = v_scrollbar->width();

259  int par_width = _m->parent->width();
260  int par_height = _m->parent->height();
261  int new_width, new_height;
262 
263  if(!v_scrollbar->isVisible()) sb_width = 0;
264  if(!h_scrollbar->isVisible()) sb_height = 0;
265 
266  new_width = par_width - (sb_width + 135);
267  new_height = par_height - (sb_height + 40);
268 
269  this->setGeometry(new_width, new_height, 130, 35);
270 
271  _m->line_edit->setFocus();
272  _m->line_edit->setText(text);
273 
274  this->show();
275 }
