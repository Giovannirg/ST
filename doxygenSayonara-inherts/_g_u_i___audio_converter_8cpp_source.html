82  }

85  ui->cb_quality->setCurrentIndex(idx);

87 
88  PlayManager* play_manager = PlayManager::getInstance();
89  connect(play_manager, &PlayManager::sig_playstate_changed, this, &GUI_AudioConverter::playstate_changed);

244 
245  if(dir.size() > 0) {
246  _settings->set(Set::Engine_CovertTargetPath, dir);
247  pl_mode_backup();
