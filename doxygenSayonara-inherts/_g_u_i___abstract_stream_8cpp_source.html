319 void GUI_AbstractStream::save_clicked()
320 {
321  QString name = _m->combo_stream->currentText();
322  QString url = _m->le_url->text();
323  GlobalMessage::Answer answer;
324 
325  if(name.isEmpty() || url.isEmpty()){
326  return;
327  }
328 
329  StreamMap map;
330 
331  for(int i=0; i<_m->combo_stream->count(); i++)
332  {
333  QString text = _m->combo_stream->itemText(i);
334  if(text == name){
335  answer = Message::question_yn(tr("Overwrite?") + "\n" + name + "\n" + url);
336  if(answer == GlobalMessage::Answer::Yes){
337  _m->stream_handler->update_url(name, url);
338  _m->stream_handler->get_all_streams(map);
339  setup_stations(map);
340  return;
341  }
342 
343  else{
344  return;
345  }
346  }
347  }
348 
349  _m->stream_handler->save(name, url);
350  _m->stream_handler->get_all_streams(map);
351  setup_stations(map);
352 }
