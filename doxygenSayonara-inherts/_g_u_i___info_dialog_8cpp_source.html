167 void GUI_InfoDialog::tab_index_changed(GUI_InfoDialog::Tab idx)
168 {
169  if(!ui){
170  return;
171  }
172 
173  ui->ui_info_widget->hide();
174  _m->ui_lyrics->hide();
175  _m->ui_tag_edit->hide();
176 
177  switch(idx)
178  {
179  case GUI_InfoDialog::Tab::Edit:
180 
181  ui->tab_widget->setCurrentWidget(_m->ui_tag_edit);
182  {
183  MetaDataList local_md;
184  for(const MetaData& md : _m->v_md){
185  if(!Helper::File::is_www(md.filepath())){
186  local_md << md;
187  }
188  }
189 
190  if(local_md.size() > 0){
191  _m->ui_tag_edit->get_tag_edit()->set_metadata(local_md);
192  }
193  }
194 
195  _m->ui_tag_edit->show();
196  break;
197 
198  case GUI_InfoDialog::Tab::Lyrics:
199 
200  ui->tab_widget->setCurrentWidget(_m->ui_lyrics);
201 
202  _m->ui_lyrics->set_metadata(_m->v_md.first());
203  _m->ui_lyrics->show();
204 
205  break;
206 
207  default:
208  ui->tab_widget->setCurrentWidget(ui->ui_info_widget);
209  ui->ui_info_widget->show();
210  prepare_cover(_m->cl);
211  break;
212  }
213 }

216 void GUI_InfoDialog::show(GUI_InfoDialog::Tab tab)
217 {
218  if(!ui){
219  init();
220  }
221 
222  QTabWidget* tab_widget = ui->tab_widget;
223 
224  prepare_cover(CoverLocation::getInvalidLocation());
225 
226  bool lyric_enabled = (_m->v_md.size() == 1);
227  bool tag_edit_enabled = std::any_of(_m->v_md.begin(), _m->v_md.end(), [](const MetaData& md){
228  return (!Helper::File::is_www(md.filepath()));
229  });
230 
231  tab_widget->setTabEnabled((int) GUI_InfoDialog::Tab::Edit, tag_edit_enabled);
232  tab_widget->setTabEnabled((int) GUI_InfoDialog::Tab::Lyrics, lyric_enabled);
233 
234  if( !tab_widget->isTabEnabled((int) tab) )
235  {
236  tab = GUI_InfoDialog::Tab::Info;
237  }
238 
239  if(tab == GUI_InfoDialog::Tab::Edit)
240  {
241  MetaDataList local_md;
242  for(const MetaData& md : _m->v_md){
243  if(!Helper::File::is_www(md.filepath())){
244  local_md << md;
245  }
246  }
247 
248  if(local_md.size() > 0){
249  _m->ui_tag_edit->get_tag_edit()->set_metadata(local_md);
250  }
251  }
252 
253  if(tab == GUI_InfoDialog::Tab::Lyrics)
254  {
255  _m->ui_lyrics->set_metadata(_m->v_md.first());
256  }
257 
258  tab_widget->setCurrentIndex((int) tab);
259  tab_index_changed(tab);
260 
261  SayonaraDialog::show();
262 }
