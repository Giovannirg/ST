38 {
39  // If an url leads me to some website content and I have to parse it
40  // and this Url is found again during parsing, it cannot be a stream
41  // and so, it cannot be a metadata object

53  bool is_url_forbidden(const QUrl& url) const
54  {
55  for(const QString& forbidden_url_str : forbidden_urls)
56  {
57  QUrl forbidden_url(forbidden_url_str);
58  QString forbidden_host = forbidden_url.host();
59  if ((forbidden_host.compare(url.host(), Qt::CaseInsensitive) == 0) &&
60  (forbidden_url.port(80) == url.port(80)) &&
61  (forbidden_url.path().compare(url.path()) == 0) &&
62  (forbidden_url.fileName().compare(url.path()) == 0))
63  {
64  return true;
65  }
66  }
67 
68  return false;
69  }
70 };

104 bool StreamParser::parse_next_url()
105 {
106  if(_m->stopped){
107  emit sig_stopped();
108  return false;
109  }
110 
111  if(_m->urls.isEmpty()) {
112  sp_log(Log::Debug, this) << "No more urls to parse";
113  emit sig_finished( _m->v_md.size() > 0);
114  return false;
115  }
116 
117  QString url = _m->urls.takeFirst();
118  AsyncWebAccess* awa = new AsyncWebAccess(this);
119  awa->set_behavior(AsyncWebAccess::Behavior::AsSayonara);
120  connect(awa, &AsyncWebAccess::sig_finished, this, &StreamParser::awa_finished);
121  awa->run(url, 5000);
122  _m->active_awa = awa;
123 
124  return true;
125 }

128 void StreamParser::awa_finished()
129 {
130  AsyncWebAccess* awa = static_cast<AsyncWebAccess*>(sender());
131  AsyncWebAccess::Status status = awa->status();
132  _m->last_url = awa->url();
133  _m->active_awa = nullptr;
134 
135  if(_m->stopped){
136  awa->deleteLater();
137  emit sig_stopped();
138  return;
139  }
140 
141  switch(status)
142  {
143  case AsyncWebAccess::Status::GotData:
144  {
145  _m->forbidden_urls << _m->last_url;
146  sp_log(Log::Warning, this) << "Got data. Try to parse content";
147 
148  QPair<MetaDataList, PlaylistFiles> result = parse_content(awa->data());
149 
150  _m->v_md << result.first;
151  _m->urls << result.second;
152 
153  _m->v_md.remove_duplicates();
154  _m->urls.removeDuplicates();
155 
156  for(MetaData& md : _m->v_md) {
157  tag_metadata(md, _m->last_url, _m->cover_url);
158  }
159  } break;
160 
161  case AsyncWebAccess::Status::NoHttp:
162  {
163  sp_log(Log::Warning, this) << "No correct http was found. Maybe Icy?";
164  IcyWebAccess* iwa = new IcyWebAccess(this);
165  _m->active_icy = iwa;
166  connect(iwa, &IcyWebAccess::sig_finished, this, &StreamParser::icy_finished);
167  iwa->check(QUrl(_m->last_url));
168 
169  awa->deleteLater();
170  } return;
171 
172  case AsyncWebAccess::Status::AudioStream:
173  {
174  sp_log(Log::Warning, this) << "Found audio stream";
175  MetaData md;
176  tag_metadata(md, _m->last_url, _m->cover_url);
177 
178  _m->v_md << md;
179  _m->v_md.remove_duplicates();
180  } break;
181 
182  default:
183  sp_log(Log::Warning, this) << "Web Access finished: " << (int) status;
184  }
185 
186  awa->deleteLater();
187 
188  if(_m->urls.size() > _m->MaxSizeUrls){
189  emit sig_too_many_urls_found(_m->urls.size(), _m->MaxSizeUrls);
190  }
191 
192  else {
193  parse_next_url();
194  }
195 }

250 QPair<MetaDataList, PlaylistFiles> StreamParser::parse_website(const QByteArray& arr) const
251 {
252  MetaDataList v_md;
253  QStringList playlist_files;
254 
255  QStringList valid_extensions;
256  valid_extensions << Helper::get_soundfile_extensions(false);
257  valid_extensions << Helper::get_playlistfile_extensions(false);
258 
259  QStringList found_strings;
260  QString re_prefix = "(http[s]*://|\"/|'/)";
261  QString re_path = "\\S+\\.(" + valid_extensions.join("|") + ")";
262  QString re_string = "(" + re_prefix + re_path + ")";
263 
264  QRegExp reg_exp(re_string);
265  QUrl parent_url(_m->last_url);
266 
267  QString website = QString::fromUtf8(arr);
268  int idx = reg_exp.indexIn(website);
269  while(idx >= 0)
270  {
271  QStringList found_urls = reg_exp.capturedTexts();
272  for(QString str : found_urls) {
273  QUrl found_url(str);
274  if( (str.size() > 7) &&
275  (!_m->is_url_forbidden(QUrl(found_url))) )
276  {
277  if(str.startsWith("\"") || str.startsWith("'")){
278  str.remove(0, 1);
279  }
280 
281  found_strings << str;
282  }
283  }
284 
285  idx = reg_exp.indexIn(website, idx + 1);
286  }
287 
288  found_strings.removeDuplicates();
289 
290  for(const QString& found_str : found_strings) {
291  QString child_url;
292 
293  if(!found_str.startsWith("http")) {
294  child_url = parent_url.scheme() + "://" + parent_url.host();
295  if(!found_str.startsWith("/")) {
296  child_url += "/";
297  }
298  }
299 
300  child_url += found_str;
301 
302  if(Helper::File::is_soundfile(found_str)){
303  MetaData md;
304  tag_metadata(md, child_url);
305  v_md << md;
306  }
307 
308  else if(Helper::File::is_playlistfile(found_str)){
309  playlist_files << child_url;
310  }
311  }
312 
313  sp_log(Log::Debug, this) << "Found " << _m->urls.size() << " playlists and " << v_md.size() << " streams";
314 
315  return QPair<MetaDataList, PlaylistFiles>(v_md, playlist_files);
316 }
