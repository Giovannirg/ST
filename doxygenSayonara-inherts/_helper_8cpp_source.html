107 QString Helper::cvt_ms_to_string(quint64 msec, bool empty_zero, bool colon, bool show_days)
108 {
109  if(msec == 0 && empty_zero){
110  return "";
111  }
112 
113  bool show_hrs = false;
114 
115  quint64 sec = msec / 1000;
116  quint64 min = sec / 60;
117 
118  quint64 secs = sec % 60;
119  quint64 hrs = min / 60;
120  quint64 days = hrs / 24;
121 
122  QString final_str;
123 
124  if(days > 0 && show_days) {
125  final_str += QString::number(days) + Lang::get(Lang::DaysShort) + " ";
126  hrs = hrs % 24;
127  show_hrs = true;
128  }
129 
130  if(!show_days) {
131  hrs += (days * 24);
132  }
133 
134  if(hrs > 0 || show_hrs) {
135  final_str += QString::number(hrs) + Lang::get(Lang::HoursShort) + " ";
136  min = min % 60;
137  }
138 
139  if(colon){
140  final_str += cvtNum2String(min, 2) + ":" + cvtNum2String(secs, 2);
141  }
142 
143  else{
144  final_str += cvtNum2String(min, 2) + Lang::get(Lang::MinutesShort) + " " + cvtNum2String(secs, 2);
145  }
146 
147  return final_str;
148 }

313  ret = new_tag;

315  QString str2search_start = QString("<") + ret + QString(".*>");
