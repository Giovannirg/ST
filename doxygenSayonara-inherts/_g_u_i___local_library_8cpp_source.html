375 void GUI_LocalLibrary::import_dirs_requested()
376 {
377  QStringList dirs;
378 
379  QFileDialog* dialog = new QFileDialog(this);
380  dialog->setDirectory(QDir::homePath());
381  dialog->setWindowTitle(Lang::get(Lang::ImportDir));
382  dialog->setFileMode(QFileDialog::DirectoryOnly);
383  dialog->setOption(QFileDialog::DontUseNativeDialog, true);
384  QList<QUrl> sidebar_urls = dialog->sidebarUrls();
385 
386  QList<QStandardPaths::StandardLocation> locations;
387  locations << QStandardPaths::HomeLocation;
388  locations << QStandardPaths::DesktopLocation;
389  locations << QStandardPaths::DownloadLocation;
390  locations << QStandardPaths::MusicLocation;
391  locations << QStandardPaths::TempLocation;
392 
393  for(QStandardPaths::StandardLocation location : locations)
394  {
395  QStringList std_locations = QStandardPaths::standardLocations(location);
396  for(const QString& std_location : std_locations){
397  QUrl url = QUrl::fromLocalFile(std_location);
398  if(sidebar_urls.contains(url)){
399  continue;
400  }
401 
402  sidebar_urls << url;
403  }
404  }
405 
406  dialog->setSidebarUrls(sidebar_urls);
407 
408  QListView* list_view = dialog->findChild<QListView*>("listView");
409  if(list_view == nullptr)
410  {
411  delete dialog;
412 
413  QString dir = QFileDialog::getExistingDirectory(this, Lang::get(Lang::ImportDir),
414  QDir::homePath(), QFileDialog::ShowDirsOnly);
415  if(!dir.isEmpty()){
416  dirs << dir;
417  }
418  }
419 
420  else{
421  list_view->setSelectionMode(QAbstractItemView::MultiSelection);
422  QTreeView* tree_view = dialog->findChild<QTreeView*>();
423  if(tree_view){
424  tree_view->setSelectionMode(QAbstractItemView::MultiSelection);
425  }
426 
427  if(dialog->exec() == QFileDialog::Accepted){
428  dirs = dialog->selectedFiles();
429  }
430  }
431 
432  if(!dirs.isEmpty()){
433  _library->import_files(dirs);
434  }
435 }
