407 static void build_genre_node(GenreNode* node, const QMap<QString, StringSet>& parent_nodes)
408 {
409  QString value = node->data;
410  if(!parent_nodes.contains(value)){
411  return;
412  }
413 
414  const StringSet& children = parent_nodes[value];
415  if(children.isEmpty()){
416  return;
417  }
418 
419  for(const QString& str : children){
420  GenreNode* new_child = new GenreNode(str);
421  build_genre_node( new_child, parent_nodes );
422  node->add_child(new_child);
423  }

427 void LibraryGenreView::init_data(const QStringList& genres)
428 {
429  SP::Set<QString> genre_set;
430  if(_m->genres){
431  delete _m->genres;
432  }
433 
434  _m->genres = new GenreNode("");
435  QMap<QString, StringSet> children;
436 
437  for(const QString& s : genres) {
438  genre_set.insert(s);
439  }
440 
441  for(auto it=genre_set.begin(); it != genre_set.end(); it++)
442  {
443  bool found_parent = false;
444  Genre genre(*it);
445  if(genre.name().isEmpty()){
446  continue;
447  }
448 
449  if(_m->toggle_tree_action->isChecked())
450  {
451  for(auto subit=genre_set.begin(); subit != genre_set.end(); subit++)
452  {
453  Genre parent_genre(*subit);
454 
455  if( parent_genre.name().isEmpty() ||
456  parent_genre == genre)
457  {
458  continue;
459  }
460 
461  if( genre.name().contains(parent_genre.name(), Qt::CaseInsensitive) ) {
462  StringSet& child_genres = children[parent_genre.name()];
463  child_genres.insert(genre.name());
464  found_parent = true;
465  }
466  }
467  }
468 
469  if(!found_parent) {
470  _m->genres->add_child(genre.name());
471  }
472  }
473 
474  for(GenreNode* base_genre : _m->genres->children){
475  build_genre_node(base_genre, children);
476  }
477 
478  _m->genres->sort(true);
479 }
