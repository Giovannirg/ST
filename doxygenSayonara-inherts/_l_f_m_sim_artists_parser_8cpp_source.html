50 void LFMSimArtistsParser::parse_document()
51 {
52  bool success;
53  QDomDocument doc("similar_artists");
54  success = doc.setContent(_data);
55 
56  if(!success){
57  _artist_match = ArtistMatch("");
58  sp_log(Log::Warning) << "Cannot parse similar artists document";
59  return;
60  }
61 
62  _artist_match = ArtistMatch(_artist_name);
63 
64  QDomElement docElement = doc.documentElement();
65  QDomNode similar_artists = docElement.firstChild(); // similarartists
66 
67  if(!similar_artists.hasChildNodes()) {
68  return;
69  }
70 
71  QString artist_name, mbid;
72  double match = -1.0;
73 
74  QDomNodeList child_nodes = similar_artists.childNodes();
75  for(int idx_artist=0; idx_artist < child_nodes.size(); idx_artist++) {
76  QDomNode artist = child_nodes.item(idx_artist);
77  QString node_name = artist.nodeName();
78 
79  if(node_name.compare("artist", Qt::CaseInsensitive) != 0) {
80  continue;
81  }
82 
83  if(!artist.hasChildNodes()) {
84  continue;
85  }
86 
87  QDomNodeList artist_child_nodes = artist.childNodes();
88  for(int idx_content = 0; idx_content <artist_child_nodes.size(); idx_content++) {
89  QDomNode content = artist_child_nodes.item(idx_content);
90  QString node_name = content.nodeName().toLower();
91 
92  if(node_name.compare("name") == 0) {
93  QDomElement e = content.toElement();
94  if(!e.isNull()) {
95  artist_name = e.text();
96  }
97  }
98 
99  else if(node_name.compare("match") == 0) {
100  QDomElement e = content.toElement();
101  if(!e.isNull()) {
102  match = e.text().toDouble();
103  }
104  }
105 
106  else if(node_name.compare("mbid") == 0){
107  QDomElement e = content.toElement();
108  if(!e.isNull()) {
109  mbid = e.text();
110  }
111  }
112 
113  if(!artist_name.isEmpty() && match > 0 && !mbid.isEmpty()) {
114  ArtistMatch::ArtistDesc artist_desc(artist_name, mbid);
115  _artist_match.add(artist_desc, match);
116  artist_name = "";
117  match = -1.0;
118  mbid = "";
119  break;
120  }
121  }
122  }
123 }
