182 void StreamWriter::data_available()
183 {
184  StreamHttpParser::HttpAnswer answer = parse_message();
185  QString ip = get_ip();
186  bool success;
187  bool close_connection = true;
188  _m->type = StreamWriter::Type::Standard;
189 
190  switch(answer){
191  case StreamHttpParser::HttpAnswer::Fail:
192  case StreamHttpParser::HttpAnswer::Reject:
193 
194  _m->type = StreamWriter::Type::Invalid;
195  //sp_log(Log::Debug, this) << "Rejected: " << _parser->get_user_agent() << ": " << get_ip();
196  send_header(true);
197  break;
198 
199  case StreamHttpParser::HttpAnswer::Ignore:
200  //sp_log(Log::Debug, this) << "ignore...";
201  break;
202 
203  case StreamHttpParser::HttpAnswer::Playlist:
204  //sp_log(Log::Debug, this) << "Asked for playlist";
205  send_playlist();
206  break;
207 
208  case StreamHttpParser::HttpAnswer::HTML5:
209  //sp_log(Log::Debug, this) << "Asked for html5";
210  send_html5();
211 
212  break;
213 
214  case StreamHttpParser::HttpAnswer::BG:
215  //sp_log(Log::Debug, this) << "Asked for background";
216  send_bg();
217  break;
218 
219  case StreamHttpParser::HttpAnswer::Favicon:
220  //sp_log(Log::Debug, this) << "Asked for favicon";
221  send_favicon();
222  break;
223 
224  case StreamHttpParser::HttpAnswer::MetaData:
225  //sp_log(Log::Debug, this) << "Asked for metadata";
226  send_metadata();
227  break;
228 
229  default:
230  _m->type = StreamWriter::Type::Streaming;
231  close_connection = false;
232  //sp_log(Log::Debug, this) << "Accepted: " << _parser->get_user_agent() << ": " << ip;
233  success = send_header(false);
234 
235  if(success){
236  _m->send_data = true;
237  _m->engine->register_raw_sound_receiver(this);
238  }
239 
240  emit sig_new_connection(ip);
241  break;
242  }
243 
244  if(close_connection){
245  _m->socket->close();
246  }
247 }
