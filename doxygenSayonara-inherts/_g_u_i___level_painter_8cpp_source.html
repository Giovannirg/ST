131 void GUI_LevelPainter::paintEvent(QPaintEvent* e) {
132  Q_UNUSED(e)
133 
134  QPainter painter(this);
135 
136  int n_rects = _cur_style.n_rects;
137  int border_x = _cur_style.hor_spacing;
138  int border_y = _cur_style.ver_spacing;
139  int n_fading_steps = _cur_style.n_fading_steps;
140  int h_rect = _cur_style.rect_height;
141  int w_rect = _cur_style.rect_width;
142 
143  int y = 10;
144  int num_zero = 0;
145  int x_init = (w_rect + border_x);
146 
147  for(int c=0; c<2; c++) {
148  float level= -std::max(_level[c], -39.9f) * 15.0f;
149  int idx = std::max(0, std::min(599, (int) level));
150 
151  level = _exp_lot[idx];
152 
153  int n_colored_rects = n_rects * level;
154 
155  QRect rect(0, y, w_rect, h_rect);
156  for(int r=0; r<n_rects; r++) {
157  if(r < n_colored_rects) {
158  if(!_cur_style.style[r].contains(-1)){
159  sp_log(Log::Debug, this) << "Style does not contain -1";
160  }
161 
162  painter.fillRect(rect, _cur_style.style[r].value(-1) );
163 
164  _steps[c][r] = n_fading_steps - 1;
165  }
166 
167  else{
168  if(!_cur_style.style[r].contains(_steps[c][r])){
169  sp_log(Log::Debug, this) << "2 Style does not contain " << _steps[c][r] << ", " << c << ", " << r;
170  }
171 
172  painter.fillRect(rect, _cur_style.style[r].value(_steps[c][r]) );
173 
174  if(_steps[c][r] > 0) {
175  _steps[c][r] -= 1;
176  }
177 
178  if(_steps[c][r] == 0) {
179  num_zero ++;
180  }
181  }
182 
183  rect.translate(x_init, 0);
184  }
185 
186  if(num_zero == 2 * n_rects) {
187  _timer->stop();
188  _timer_stopped = true;
189  }
190 
191  y += h_rect + border_y;
192  }
193 }
