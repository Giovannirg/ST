76  }
77  }
78 
79  v_md_tmp = playlist_parser->tracks();
80 
81  for(const MetaData& md : v_md_tmp) {
82  if( Helper::File::check_file(md.filepath()) ){
83  result << md;
84  }
85

87  v_md_to_delete << md;
88  }
89  }
