245 void TagEdit::run()
246 {
247  MetaDataList v_md;
248  MetaDataList v_md_orig;
249  DatabaseConnector* db;
250 
251  sp_log(Log::Debug, this) << "Apply albums and artists";
252  apply_artists_and_albums_to_md();
253 
254  sp_log(Log::Debug, this) << "Have to change" <<
255  std::count(_m->changed_md.begin(), _m->changed_md.end(), true)
256  << " tracks";
257 
258  int i=0;
259  int n_operations = _m->v_md.size() + _m->cover_map.size();
260  for(i=0; i<_m->v_md.size(); i++)
261  {
262  MetaData md = _m->v_md[i];
263  emit sig_progress( (i * 100) / n_operations);
264 
265  if( _m->changed_md[i] == false ) {
266  continue;
267  }
268 
269  bool success = Tagging::setMetaDataOfFile(md);
270  if( !success ) {
271  continue;
272  }
273 
274  if( !md.is_extern && md.id >= 0 ){
275  success = _m->ldb->updateTrack(md);
276  }
277 
278  if(success){
279  v_md << std::move(md);
280  v_md_orig.push_back(_m->v_md_orig[i]);
281  }
282  }
283 
284  for(int idx : _m->cover_map.keys()){
285 
286  Tagging::write_cover(_m->v_md[idx], _m->cover_map[idx]);
287  emit sig_progress( (i++ * 100) / n_operations);
288  }
289 
290  _m->ldb->createIndexes();
291 
292  db = DatabaseConnector::getInstance();
293  db->clean_up();
294 
295  _m->v_md_after_change = v_md;
296  _m->v_md_before_change = v_md_orig;
297  _m->v_md_orig = _m->v_md;
298 
299  emit sig_progress(-1);
300 }
