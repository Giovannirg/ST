72 {

74  const int PauseBetweenRequests = 300;
75  const int NumParallelRequests = 10;
76 
77  while(_m->may_run) {
78 
79  while(_m->hashes.isEmpty() || !_m->goon) {
80  Helper::sleep_ms(PauseBetweenRequests);

82 
83  _m->goon = false;
84

88 
89  for(int i=0; i<NumParallelRequests; i++){
90  if(_m->hashes.isEmpty()) {
91  break;
92  }
93 
94  if(i > 0) {
95  Helper::sleep_ms(PauseBetweenRequests);
96  }
97 
98  try{
99  std::lock_guard<std::mutex> guard(_m->mutex);
100  _m->current_hash = _m->hashes.takeFirst();
101  _m->current_cl = _m->cover_locations.takeFirst();
102 
103  emit sig_next();
104 
105  } catch(std::exception* e) {
