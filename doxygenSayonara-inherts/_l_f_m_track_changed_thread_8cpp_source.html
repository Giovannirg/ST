91  _m->md = md;

93  LFMWebAccess* lfm_wa = new LFMWebAccess();

95  connect(lfm_wa, &LFMWebAccess::sig_response, this, &LFMTrackChangedThread::response_update);
96  connect(lfm_wa, &LFMWebAccess::sig_error, this, &LFMTrackChangedThread::error_update);
97 
98  QString artist = _m->md.artist;
99  QString title = _m->md.title;
100 
101  if(artist.trimmed().size() == 0) artist = "Unknown";
102  artist.replace("&", "&amp;");
103 
104  UrlParams sig_data;
105  sig_data["api_key"] = LFM_API_KEY;
106  sig_data["artist"] = artist.toLocal8Bit();
107  sig_data["duration"] = QString::number(_m->md.length_ms / 1000).toLocal8Bit();
108  sig_data["method"] = QString("track.updatenowplaying").toLocal8Bit();
109  sig_data["sk"] = _m->session_key.toLocal8Bit();
110  sig_data["track"] = title.toLocal8Bit();
111 
112  sig_data.append_signature();
113 
114 
115  QByteArray post_data;
116  QString url = lfm_wa->create_std_url_post(
117  QString("http://ws.audioscrobbler.com/2.0/"),
118  sig_data,
119  post_data);
120 
121  lfm_wa->call_post_url(url, post_data);
122 }

142 void LFMTrackChangedThread::search_similar_artists(const MetaData& md) {
143  if(md.db_id != 0) {
144  return;
145  }
146 
147  if(md.radio_mode() == RadioMode::Station){
148  return;
149  }
150 
151  _m->md = md;
152 
153  if(_m->md.artist.trimmed().isEmpty()){
154  return;
155  }
156 
157  // check if already in cache
158  if(_m->sim_artists_cache.contains(_m->md.artist)) {
159  const ArtistMatch& artist_match = _m->sim_artists_cache.value(_m->md.artist);
160  evaluate_artist_match(artist_match);
161  return;
162  }
163 
164  _m->artist = _m->md.artist;
165 
166  LFMWebAccess* lfm_wa = new LFMWebAccess();
167 
168  connect(lfm_wa, &LFMWebAccess::sig_response, this, &LFMTrackChangedThread::response_sim_artists);
169  connect(lfm_wa, &LFMWebAccess::sig_error, this, &LFMTrackChangedThread::error_sim_artists);
170 
171  QString url = QString("http://ws.audioscrobbler.com/2.0/?");
172  QString encoded = QUrl::toPercentEncoding( _m->md.artist );
173  url += QString("method=artist.getsimilar&");
174  url += QString("artist=") + encoded + QString("&");
175  url += QString("api_key=") + LFM_API_KEY;
176 
177  lfm_wa->call_url(url);
178  return;
179 }

182 void LFMTrackChangedThread::evaluate_artist_match(const ArtistMatch& artist_match){
183  if(!artist_match.is_valid()){
184  return;
185  }
186 
187  QByteArray arr = Compressor::compress(artist_match.to_string().toLocal8Bit());
188  Helper::File::create_directories(Helper::get_sayonara_path() + "/similar_artists/");
189  Helper::File::write_file(arr, Helper::get_sayonara_path() + "/similar_artists/" + artist_match.get_artist_name() + ".comp");
190 
191  // if we always take the best, it's boring
192  ArtistMatch::Quality quality, quality_org;
193 
194  int rnd_number = Helper::get_random_number(1, 999);
195 
196  if(rnd_number > 350) {
197  quality = ArtistMatch::Quality::Very_Good; // [250-999]
198  }
199 
200  else if(rnd_number > 75){
201  quality = ArtistMatch::Quality::Well; // [50-250]
202  }
203 
204  else {
205  quality = ArtistMatch::Quality::Poor;
206  }
207 
208  quality_org = quality;
209  QMap<QString, int> possible_artists;
210 
211  while(possible_artists.isEmpty()) {
212  possible_artists = filter_available_artists(artist_match, quality);
213 
214  switch(quality){
215  case ArtistMatch::Quality::Poor:
216  quality = ArtistMatch::Quality::Very_Good;
217  break;
218  case ArtistMatch::Quality::Well:
219  quality = ArtistMatch::Quality::Poor;
220  break;
221  case ArtistMatch::Quality::Very_Good:
222  quality = ArtistMatch::Quality::Well;
223  break;
224  default: // will never be executed
225  quality = quality_org;
226  break;
227  }
228 
229  if(quality == quality_org) {
230  break;
231  }
232  }
233 
234  if(possible_artists.isEmpty()){
235  return;
236  }
237 
238  IDList chosen_ids;
239  for(auto it = possible_artists.begin(); it != possible_artists.end(); it++) {
240  chosen_ids << it.value();
241  }
242 
243  emit sig_similar_artists_available(chosen_ids);
244 }

247 QMap<QString, int> LFMTrackChangedThread::filter_available_artists(const ArtistMatch& artist_match, ArtistMatch::Quality quality) {
248  DatabaseConnector* db = DatabaseConnector::getInstance();
249  QMap<ArtistMatch::ArtistDesc, double> bin = artist_match.get(quality);
250  QMap<QString, int> possible_artists;
251 
252  for(const ArtistMatch::ArtistDesc& key : bin.keys()) {
253 #if SMART_COMPARE
254 
255  QMap<QString, float> sc_map = _smart_comparison->get_similar_strings(key);
256  for(const QString& sc_key : sc_map.keys() ){
257  int artist_id = db->getArtistID(sc_key);
258  if(artist_id >= 0 && sc_map[sc_key] > 5.0f){
259  possible_artists[sc_key] = artist_id;
260  }
261 
262  }
263 
264 #else
265  int artist_id = db->getArtistID(key.artist_name);
266  if(artist_id >= 0 ){
267  possible_artists[key.artist_name] = artist_id;
268  }
269 
270 #endif
271  }
272 
273  return possible_artists;
274 }
