432 static Tagging::TagType tag_type_from_fileref(const TagLib::FileRef& f)
433 {
434  TagLib::MPEG::File* mpg = dynamic_cast<TagLib::MPEG::File*>(f.file());
435  if(mpg)
436  {
437  if(mpg->hasID3v2Tag()){
438  return Tagging::TagType::ID3v2;
439  }
440 
441  if(mpg->hasID3v1Tag()){
442  return Tagging::TagType::ID3v1;
443  }
444  }
445 
446  TagLib::FLAC::File* flac = dynamic_cast<TagLib::FLAC::File*>(f.file());
447  if(flac)
448  {
449  if(flac->hasID3v2Tag()){
450  return Tagging::TagType::ID3v2;
451  }
452 
453  if(flac->hasID3v1Tag()){
454  return Tagging::TagType::ID3v1;
455  }
456  }
457 
458  TagLib::Tag* tag = f.tag();
459  if(dynamic_cast<TagLib::ID3v2::Tag*>(tag) != nullptr){
460  return Tagging::TagType::ID3v2;
461  }
462 
463  if(dynamic_cast<TagLib::ID3v1::Tag*>(tag) != nullptr){
464  return Tagging::TagType::ID3v1;
465  }
466 
467  if(dynamic_cast<TagLib::Ogg::XiphComment*>(tag) != nullptr){
468  return Tagging::TagType::Xiph;
469  }
470 
471  if(dynamic_cast<TagLib::MP4::Tag*>(tag) != nullptr){
472  return Tagging::TagType::MP4;
473  }
474 
475  return Tagging::TagType::Unsupported;
476 }
