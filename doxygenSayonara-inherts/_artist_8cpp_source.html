83 QString ArtistList::get_major_artist(const QStringList& artists)
84 {
85  QHash<QString, int> map;
86  int n_artists = artists.size();
87 
88  if(artists.isEmpty()) {
89  return "";
90  }
91 
92  if(n_artists == 1){
93  return artists.first().toLower().trimmed();
94  }
95 
96  for(const QString& artist : artists) {
97  QString alower = artist.toLower().trimmed();
98 
99  // count appearance of artist
100  if( !map.contains(alower) ) {
101  map.insert(alower, 1);
102  }
103  else {
104  map[alower] = map.value(alower) + 1;
105  };
106  }
107 
108  // n_appearances have to be at least 2/3 of all appearances
109  for(const QString& artist : map.keys()) {
110  int n_appearances = map.value(artist);
111  if(n_appearances * 3 > n_artists * 2) {
112  return artist;
113  }
114 
115  }
116 
117  return Lang::get(Lang::Various);
118 }
