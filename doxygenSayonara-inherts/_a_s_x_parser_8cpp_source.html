39  QDomElement docElement = doc.documentElement();
40  QDomNode child_node = docElement.firstChild();

42 
43  QString node_name = child_node.nodeName();
44  if(node_name.compare("entry", Qt::CaseInsensitive) == 0){
45  entry = child_node.toElement();
46  }
47

51

74

77  }
78 
79  else if(!node_name.compare("album")) {
