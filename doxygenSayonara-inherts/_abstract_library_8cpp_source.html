381 void AbstractLibrary::change_album_selection(const SP::Set<int>& indexes)
382 {
383  SP::Set<AlbumID> selected_albums;
384  bool show_album_artists = _settings->get(Set::Lib_ShowAlbumArtists);
385 
386  for(auto it=indexes.begin(); it != indexes.end(); it++){
387  int idx = *it;
388  if(idx >= _vec_albums.size()){
389  break;
390  }
391 
392  const Album& album = _vec_albums[idx];
393  selected_albums.insert(album.id);
394  }
395 
396  if(selected_albums == _selected_albums) {
397  return;
398  }
399 
400  _vec_md.clear();
401  _selected_albums = selected_albums;
402 
403  // only show tracks of selected album / artist
404  if(_selected_artists.size() > 0) {
405  if(_selected_albums.size() > 0) {
406  MetaDataList v_md;
407 
408  get_all_tracks_by_album(_selected_albums.toList(), v_md, _filter, _sortorder);
409 
410  // filter by artist
411 
412  for(const MetaData& md : v_md) {
413  ArtistID artist_id;
414  if(show_album_artists){
415  artist_id = md.album_artist_id();
416  }
417 
418  else{
419  artist_id = md.artist_id;
420  }
421 
422  if(_selected_artists.contains(artist_id)){
423  _vec_md << std::move(md);
424  }
425  }
426  }
427 
428  else{
429  get_all_tracks_by_artist(_selected_artists.toList(), _vec_md, _filter, _sortorder);
430  }
431  }
432 
433  // only album is selected
434  else if(_selected_albums.size() > 0) {
435  get_all_tracks_by_album(_selected_albums.toList(), _vec_md, _filter, _sortorder);
436  }
437 
438  // neither album nor artist, but searchstring
439  else if(!_filter.cleared()) {
440  get_all_tracks_by_searchstring(_filter, _vec_md, _sortorder);
441  }
442 
443  // no album, no artist, no searchstring
444  else{
445  get_all_tracks(_vec_md, _sortorder);
446  }
447 
448  restore_track_selection();
449 }
