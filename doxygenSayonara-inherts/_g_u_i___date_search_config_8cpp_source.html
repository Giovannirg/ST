252  button_span_ot.insert(Library::DateFilter::TimeSpan::Years, ui->rb_ot_years);
253 
254  button_span_bwf.insert(Library::DateFilter::TimeSpan::Days, ui->rb_bwf_days);
255  button_span_bwf.insert(Library::DateFilter::TimeSpan::Weeks, ui->rb_bwf_weeks);
256  button_span_bwf.insert(Library::DateFilter::TimeSpan::Months, ui->rb_bwf_months);
257  button_span_bwf.insert(Library::DateFilter::TimeSpan::Years, ui->rb_bwf_years);
258 
259  button_span_bwt.insert(Library::DateFilter::TimeSpan::Days, ui->rb_bwt_days);

273 
274  ui->comboBox->setCurrentIndex(0);

278 
279 
280  case Library::DateFilter::Type::OlderThan:

284  btn->setChecked(true);
285 
286  ui->sb_ot->setValue(time_span_map[0].second);

288  ui->comboBox->setCurrentIndex(1);

291  }

298  return;

311 
312  ui->comboBox->setCurrentIndex(2);
313  ui->stackedWidget->setCurrentIndex(2);
314  }
315 
316  break;
317 
318  default:
319  break;
320  }
321 
322  _m->org_filter = filter;
323  _m->edited_filter = Library::DateFilter();
324 
325  ui->le_title->setText(filter.name());
