140 void GUI_Broadcast::connection_closed(const QString& ip)
141 {

145

147  return;

149 
150  sp_log(Log::Info) << "Connection closed: " << ip;
151 
152  int idx;
153  for(idx=0; idx<ui->combo_clients->count(); idx++){
154  if(ui->combo_clients->itemText(idx).contains(ip)){
155  break;
156  }
157  }
158 
159  if(idx >= ui->combo_clients->count()){
160  return;
161  }
162 
163  ui->combo_clients->removeItem(idx);
164 
165  if(ui->combo_clients->count() == 0){
166  ui->btn_dismiss->setEnabled(false);
