274 void PlaylistTabBar::dropEvent(QDropEvent* e)
275 {
276  e->accept();
277  int tab = this->tabAt(e->pos());
278 
279  _drag_origin_tab = _tab_before_dd;
280 
281  if(_tab_before_dd >= 0 && currentIndex() == count() - 1){
282  this->setCurrentIndex(_tab_before_dd);
283  }
284 
285  _tab_before_dd = -1;
286 
287  const QMimeData* mime_data = e->mimeData();
288  if(!mime_data){
289  return;
290  }
291 
292  const CustomMimeData* cmd = dynamic_cast<const CustomMimeData*>(mime_data);
293  if(!cmd){
294  if(!mime_data->hasUrls()){
295  return;
296  }
297 
298  MetaDataList v_md;
299  DirectoryReader dir_reader;
300  QList<QUrl> urls = mime_data->urls();
301  QStringList files;
302  for(const QUrl& url : urls){
303  files << url.toLocalFile();
304  }
305 
306  v_md = dir_reader.get_md_from_filelist(files);
307  emit sig_metadata_dropped(tab, v_md);
308  return;
309  }
310 
311  if(!cmd->hasMetaData()){
312  return;
313  }
314 
315  MetaDataList v_md = cmd->getMetaData();
316 
317  emit sig_metadata_dropped(tab, v_md);
318 }
